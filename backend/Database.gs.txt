function getSheet(name) {
  const ss = getConnection();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    // Configuración inicial de cabeceras si se crean hojas nuevas
    if(name === 'antecedentes_medicos') {
      sheet.appendRow(['id_paciente', 'antecedentes', 'fecha_registro']);
    }
  }
  return sheet;
}

// --- HELPER MAESTRO DE FECHAS (CORE FIX) ---
// Convierte cualquier formato de fecha de Excel a "YYYY-MM-DD"
function toIsoDate(val) {
  if (!val) return "";
  
  // Si es un Objeto Fecha (lo que devuelve Sheets)
  if (val instanceof Date) {
    const y = val.getFullYear();
    const m = String(val.getMonth() + 1).padStart(2, '0'); // Mes base 0
    const d = String(val.getDate()).padStart(2, '0');
    return `${y}-${m}-${d}`;
  }
  
  // Si ya es Texto (ej: "2026-01-14T00:00...")
  return String(val).split('T')[0];
}

// Helper para formatear hora visualmente
function formatTime(val) {
  if (!val) return "";
  if (val instanceof Date) {
      let h = val.getHours();
      let m = val.getMinutes();
      return (h<10?'0':'')+h + ':' + (m<10?'0':'')+m;
  }
  return String(val); 
}

function isPublicFileLinksEnabled_() {
  // Modo compatible: por defecto se mantiene enlace publico.
  // Para endurecer sin tocar codigo, definir Script Property:
  // VIDAFEM_PUBLIC_FILE_LINKS = "false"
  try {
    const raw = PropertiesService.getScriptProperties().getProperty("VIDAFEM_PUBLIC_FILE_LINKS");
    const val = String(raw === null || raw === undefined ? "" : raw).trim().toLowerCase();
    if (val === "false" || val === "0" || val === "no" || val === "off") return false;
    return true;
  } catch (e) {
    return true;
  }
}

function applyFileSharingPolicy_(file) {
  if (!file) return;
  if (!isPublicFileLinksEnabled_()) return;
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
}

function getSupabaseSyncConfig_() {
  try {
    const props = PropertiesService.getScriptProperties();
    const enabledRaw = String(props.getProperty("VIDAFEM_SUPABASE_SYNC_ENABLED") || "").trim().toLowerCase();
    const url = String(props.getProperty("VIDAFEM_SUPABASE_URL") || "").trim().replace(/\/+$/, "");
    const serviceRole = String(props.getProperty("VIDAFEM_SUPABASE_SERVICE_ROLE") || "").trim();
    const disabled = enabledRaw === "false" || enabledRaw === "0" || enabledRaw === "no" || enabledRaw === "off";
    return {
      enabled: !disabled && !!url && !!serviceRole,
      url: url,
      serviceRole: serviceRole
    };
  } catch (e) {
    return { enabled: false, url: "", serviceRole: "" };
  }
}

function supabaseRestRequest_(method, tableName, options) {
  const cfg = getSupabaseSyncConfig_();
  if (!cfg.enabled) {
    return { success: false, skipped: true, message: "Sync Supabase desactivado." };
  }

  const opts = options || {};
  const query = [];
  if (opts.onConflict) {
    query.push("on_conflict=" + encodeURIComponent(String(opts.onConflict)));
  }
  if (opts.select) {
    query.push("select=" + encodeURIComponent(String(opts.select)));
  }
  if (opts.filters) {
    Object.keys(opts.filters).forEach(function(key) {
      const value = String(opts.filters[key] === undefined || opts.filters[key] === null ? "" : opts.filters[key]).trim();
      if (!key || !value) return;
      query.push(encodeURIComponent(key) + "=" + encodeURIComponent("eq." + value));
    });
  }

  const url = cfg.url + "/rest/v1/" + encodeURIComponent(String(tableName || "").trim()) + (query.length ? ("?" + query.join("&")) : "");
  const headers = {
    "apikey": cfg.serviceRole,
    "Authorization": "Bearer " + cfg.serviceRole,
    "Accept": "application/json"
  };
  if (opts.prefer) headers["Prefer"] = opts.prefer;

  const params = {
    method: String(method || "get").toLowerCase(),
    headers: headers,
    muteHttpExceptions: true
  };

  if (hasOwnKey_(opts, "body")) {
    params.contentType = "application/json";
    params.payload = JSON.stringify(opts.body);
  }

  try {
    const resp = UrlFetchApp.fetch(url, params);
    const code = resp.getResponseCode();
    const text = resp.getContentText() || "";
    let data = null;
    try { data = text ? JSON.parse(text) : null; } catch (eJson) {}
    if (code >= 200 && code < 300) {
      return { success: true, status: code, data: data };
    }
    return {
      success: false,
      status: code,
      message: text || ("HTTP " + code),
      data: data
    };
  } catch (e) {
    return { success: false, message: e.toString() };
  }
}

function normalizeServiceNamesForSupabase_(names) {
  const list = Array.isArray(names) ? names : [names];
  const seen = {};
  const out = [];
  list.forEach(function(name) {
    const clean = String(name || "").trim();
    if (!clean || seen[clean]) return;
    seen[clean] = true;
    out.push(clean);
  });
  return out;
}

function deleteServiceConfigFromSupabase_(serviceNames) {
  const names = normalizeServiceNamesForSupabase_(serviceNames);
  for (let i = 0; i < names.length; i++) {
    const res = supabaseRestRequest_("delete", "config_campos", {
      filters: { servicio: names[i] }
    });
    if (res && !res.success && !res.skipped) return res;
    if (res && res.skipped) return res;
  }
  return { success: true };
}

function deleteServiceRecordFromSupabase_(idServicio, serviceName) {
  const id = String(idServicio || "").trim();
  const name = String(serviceName || "").trim();
  if (!id && !name) return { success: false, message: "Servicio sin referencia para eliminar." };
  return id
    ? supabaseRestRequest_("delete", "servicios", { filters: { id: id } })
    : supabaseRestRequest_("delete", "servicios", { filters: { nombre_servicio: name } });
}

function syncServiceFullToSupabase_(serviceObj, fieldList, previousServiceNames) {
  const serviceRow = buildServiceSupabaseRow_(serviceObj);
  if (!serviceRow.id || !serviceRow.nombre_servicio) {
    return { success: false, message: "Servicio sin datos clave para sincronizar." };
  }

  const deleteServiceRes = deleteServiceRecordFromSupabase_(serviceRow.id, serviceRow.nombre_servicio);
  if (deleteServiceRes && !deleteServiceRes.success && !deleteServiceRes.skipped) return deleteServiceRes;
  if (deleteServiceRes && deleteServiceRes.skipped) return deleteServiceRes;

  const insertServiceRes = supabaseRestRequest_("post", "servicios", {
    prefer: "return=representation",
    body: serviceRow
  });
  if (insertServiceRes && !insertServiceRes.success && !insertServiceRes.skipped) return insertServiceRes;
  if (insertServiceRes && insertServiceRes.skipped) return insertServiceRes;

  const deleteRes = deleteServiceConfigFromSupabase_(
    normalizeServiceNamesForSupabase_((previousServiceNames || []).concat([serviceRow.nombre_servicio]))
  );
  if (deleteRes && !deleteRes.success && !deleteRes.skipped) return deleteRes;
  if (deleteRes && deleteRes.skipped) return deleteRes;

  const rows = [];
  if (Array.isArray(fieldList)) {
    fieldList.forEach(function(field) {
      const row = buildServiceFieldSupabaseRow_(serviceRow.nombre_servicio, field);
      if (!row.campo_nombre) return;
      rows.push(row);
    });
  }
  if (!rows.length) return { success: true };

  return supabaseRestRequest_("post", "config_campos", {
    prefer: "return=representation",
    body: rows
  });
}

function deleteServiceFromSupabase_(idServicio, serviceName) {
  const deleteRes = deleteServiceRecordFromSupabase_(idServicio, serviceName);
  if (deleteRes && !deleteRes.success && !deleteRes.skipped) return deleteRes;
  if (deleteRes && deleteRes.skipped) return deleteRes;

  return deleteServiceConfigFromSupabase_([serviceName]);
}

function diagnosisDateToIsoString_(value) {
  if (!value) return "";
  if (value instanceof Date) {
    try { return value.toISOString(); } catch (e) {}
  }
  const raw = String(value || "").trim();
  if (!raw) return "";
  const parsed = new Date(raw);
  if (!isNaN(parsed.getTime())) {
    try { return parsed.toISOString(); } catch (e) {}
  }
  return raw;
}

function supabaseDateText_(value) {
  if (value === undefined || value === null || value === "") return "";
  return toIsoDate(value);
}

function supabaseNullableText_(value, options) {
  const opts = options || {};
  let txt = String(value === undefined || value === null ? "" : value).trim();
  if (!txt) return null;
  if (opts.lowercase) txt = txt.toLowerCase();
  if (opts.uppercase) txt = txt.toUpperCase();
  return txt;
}

function supabaseNullableDateText_(value) {
  const txt = supabaseDateText_(value);
  return txt ? txt : null;
}

function supabaseNullableIsoString_(value) {
  const txt = diagnosisDateToIsoString_(value);
  return txt ? txt : null;
}

function normalizeSheetFlagForSupabase_(value, fallback) {
  const defaultValue = fallback === undefined ? false : !!fallback;
  if (value === undefined || value === null || value === "") return defaultValue ? "SI" : "NO";
  if (value === true || value === 1) return "SI";
  if (value === false || value === 0) return "NO";
  const raw = String(value || "").trim().toLowerCase();
  if (!raw) return defaultValue ? "SI" : "NO";
  if (raw === "si" || raw === "true" || raw === "1" || raw === "yes" || raw === "on") return "SI";
  if (raw === "no" || raw === "false" || raw === "0" || raw === "off") return "NO";
  return defaultValue ? "SI" : "NO";
}

function findRow(sheetName, colName, value) {
  const sheet = getSheet(sheetName);
  const data = sheet.getDataRange().getValues();
  if (data.length === 0) return null;
  const headers = data[0];
  const colIndex = headers.indexOf(colName);
  if (colIndex === -1) return null;

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][colIndex]) === String(value)) {
      let rowObj = {};
      for (let j = 0; j < headers.length; j++) {
        rowObj[headers[j]] = data[i][j];
      }
      return rowObj;
    }
  }
  return null;
}

function findRowIndex(sheet, colName, value) {
  const data = sheet.getDataRange().getValues();
  if (data.length === 0) return -1;
  const headers = data[0];
  const colIndex = headers.indexOf(colName);
  if (colIndex === -1) return -1;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][colIndex]) === String(value)) return i + 1;
  }
  return -1;
}

}

function createRecord(sheetName, dataObj) {
  // 1. OBTENER EL CANDADO (LOCK)
  const lock = LockService.getScriptLock();

  try {
    // Intentamos bloquear el script por 10 segundos.
    lock.waitLock(10000);
  } catch (e) {
    return { success: false, message: "El servidor está ocupado. Inténtalo de nuevo en unos segundos." };
  }

  // 2. EJECUTAR LA LÓGICA DE GUARDADO (DENTRO DEL BLOQUEO)
  try {
    const sheet = getSheet(sheetName);
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    
    // Mapeamos los datos según el orden de las columnas
    let newRow = headers.map(header => dataObj[header] || "");
    let warningMessage = "";
    
    // Insertamos la fila
    sheet.appendRow(newRow);
    
    // Forzamos formato texto para evitar que Excel cambie fechas o números
    const lastRow = sheet.getLastRow();
    const range = sheet.getRange(lastRow, 1, 1, headers.length);
    range.setNumberFormat('@'); 
    range.setValues([newRow]); 
    
    // --- LÓGICA DE CARPETAS (Se mantiene igual, pero protegida) ---
    if (sheetName === "pacientes") {
        try {
            const rootFolder = getFolder("VIDAFEM_EXPEDIENTES");
            const folderName = `${dataObj.nombre_completo}_${dataObj.id_paciente}`;
            getFolder(folderName, rootFolder);
            
        } catch(e) {
            console.log("Advertencia: No se pudo crear la carpeta inicial. " + e.toString());
        }

        const savedRow = {};
        headers.forEach(function(header, idx) {
          savedRow[header] = newRow[idx];
        });
        const syncRes = syncPatientToSupabase_(savedRow);
        if (syncRes && !syncRes.success && !syncRes.skipped) {
          warningMessage = syncRes.message || "No se pudo sincronizar con Supabase.";
        }
    }

    return warningMessage
      ? {
          success: true,
          message: "Guardado correctamente. Advertencia: no se pudo sincronizar con Supabase.",
          warning: warningMessage
        }
      : { success: true, message: "Guardado correctamente" };

  } catch (error) {
    return { success: false, message: "Error crítico: " + error.toString() };

  } finally {
    lock.releaseLock(); 
  }
}

function updateRecord(sheetName, idValue, dataObj) {
  const sheet = getSheet(sheetName);
  const rowIndex = findRowIndex(sheet, "id_paciente", idValue);
  if (rowIndex === -1) return { success: false, message: "ID no encontrado" };
  
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  headers.forEach((header, colIndex) => {
    if (dataObj[header] !== undefined) {
      const cell = sheet.getRange(rowIndex, colIndex + 1);
      cell.setNumberFormat('@'); 
      cell.setValue(dataObj[header]);
    }
  });

  if (sheetName === "pacientes") {
    const updatedRow = findRow(sheetName, "id_paciente", idValue);

    if (syncRes && !syncRes.success && !syncRes.skipped) {
      return {

        message: "Actualizado correctamente. Advertencia: no se pudo sincronizar con Supabase.",
        warning: syncRes.message || "No se pudo sincronizar con Supabase."
      };
    }
  }

  return { success: true, message: "Actualizado correctamente" };
}

    const sheetCampos = ensureServiceConfigSheetHeaders_().sheet;

function deleteRecord(sheetName, idValue) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(5000);
    const sheet = getSheet(sheetName);
    let warningMessage = "";
    
    // 1. Identificar columna ID según la hoja
    let colID = "id_paciente";
    if(sheetName === "servicios") colID = "id";
    if(sheetName === "config_promociones") colID = "id_promo";

    // 2. Buscar la fila en Excel
    const rowIndex = findRowIndex(sheet, colID, idValue);
    if (rowIndex === -1) return { success: false, message: "ID no encontrado" };
    const deletedRow = sheetName === "pacientes" ? findRow(sheetName, colID, idValue) : null;

    // ======================================================
    // 3. SI ES UN PACIENTE -> BORRADO EN CASCADA (CORREGIDO)
    // ======================================================
    if (sheetName === "pacientes") {
      
      // A. BORRAR CARPETA FÍSICA EN DRIVE (Búsqueda Inteligente por ID)
      try {
          const rootFolders = DriveApp.getFoldersByName("VIDAFEM_EXPEDIENTES");
          if(rootFolders.hasNext()) {
              const root = rootFolders.next();
              const subFolders = root.getFolders();
              const searchId = String(idValue).trim();

              // Recorremos las carpetas buscando la coincidencia de ID
              while (subFolders.hasNext()) {
                  const folder = subFolders.next();
                  // Si el nombre de la carpeta TERMINA con el ID (ej: "Juan Perez_12345")
                  if (folder.getName().endsWith(searchId)) {
                      folder.setTrashed(true); // ¡A la basura!
                      break; // Ya la encontramos, dejamos de buscar
                  }
              }
          }
      } catch(e) { 
          console.log("Error borrando carpeta Drive: " + e); 
      }

      // B. Borrar Historia Clínica
      try {
         const sheetHC = getSheet("historia_clinica");
         const idxHC = findRowIndex(sheetHC, "id_paciente", idValue);
         if (idxHC !== -1) sheetHC.deleteRow(idxHC);
      } catch(e) {}

      // C. Borrar Citas y Reportes (Usa los helpers que ya tienes)
      try { deleteRowsBulk("citas", "id_paciente", idValue); } catch(e){}
      try { deleteRowsBulk("diagnosticos_archivos", "id_paciente", idValue); } catch(e){}
    }
    
    // 4. Finalmente, borrar al paciente de la lista de Excel
    sheet.deleteRow(rowIndex);

    if (sheetName === "pacientes") {
      const syncRes = deletePatientFromSupabase_(deletedRow && deletedRow.id_paciente ? deletedRow.id_paciente : idValue);
      if (syncRes && !syncRes.success && !syncRes.skipped) {
        warningMessage = syncRes.message || "No se pudo sincronizar con Supabase.";
      }
    }
    
    return warningMessage
      ? {
          success: true,
          message: "Paciente y todos sus datos eliminados correctamente. Advertencia: no se pudo sincronizar con Supabase.",
          warning: warningMessage
        }
      : { success: true, message: "Paciente y todos sus datos eliminados correctamente." };

  } catch (error) {
    return { success: false, message: "Error al eliminar: " + error.toString() };
  } finally {
    lock.releaseLock();
  }
}

function getServices(requester) {
  const ensured = ensureServicesSheetHeaders_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];

  const idxId = headers.indexOf("id");
  const idxName = headers.indexOf("nombre_servicio");
  const idxRecs = headers.indexOf("recomendaciones");
  const idxTitle = headers.indexOf("titulo_reporte");
  const idxScope = headers.indexOf("scope_visibility");
  const idxOwner = headers.indexOf("owner_usuario");
  const idxDuration = headers.indexOf("duracion_minutos");

  const access = resolveServiceDoctorRequester_(requester);
  const filterByRequester = !!String(requester || "").trim() && !access.superadmin;
  const doctorUser = String(access.doctor || "").trim().toLowerCase();

  let result = [];
  for (let i = 1; i < data.length; i++) {
    const nombre = idxName > -1 ? String(data[i][idxName] || "").trim() : "";
    if (!nombre) continue;

    let scope = idxScope > -1 ? String(data[i][idxScope] || "").trim().toUpperCase() : "ALL";
    if (scope !== "OWNER" && scope !== "ALL") scope = "ALL";
    const owner = idxOwner > -1 ? String(data[i][idxOwner] || "").trim().toLowerCase() : "";

    const allowed = !filterByRequester || scope === "ALL" || (doctorUser && owner === doctorUser);
    if (!allowed) continue;

    result.push({
      id: idxId > -1 ? data[i][idxId] : "",
      nombre_servicio: nombre,
      recomendaciones: idxRecs > -1 ? (data[i][idxRecs] || "") : "",
      titulo_reporte: idxTitle > -1 ? (data[i][idxTitle] || "") : "",
      scope_visibility: scope,
      owner_usuario: owner,
      duracion_minutos: idxDuration > -1 ? normalizeDurationMinutes_(data[i][idxDuration]) : 30,
      duracion_label: idxDuration > -1 ? durationMinutesToLabel_(data[i][idxDuration]) : durationMinutesToLabel_(30)
    });
  }
  return result;
}

function parseAllowedDurationMinutes_(value) {
  const raw = String(value === undefined || value === null ? "" : value).trim();
  if (!raw) return 0;
  const num = Number(raw);
  const allowed = { 30: true, 60: true, 120: true, 180: true, 240: true, 300: true };
  return allowed[num] ? num : 0;
}

function normalizeDurationMinutes_(value) {
  return parseAllowedDurationMinutes_(value) || 30;
}

function durationMinutesToLabel_(value) {
  const mins = normalizeDurationMinutes_(value);
  if (mins === 30) return "30 minutos";
  const hours = mins / 60;
  return hours + " hora" + (hours === 1 ? "" : "s");
}

function timeTextToMinutes_(value) {
  if (value instanceof Date) {
    return (value.getHours() * 60) + value.getMinutes();
  }

  if (typeof value === "number" && isFinite(value)) {
    if (value >= 0 && value < 1) {
      return Math.round(value * 24 * 60);
    }
    if (value >= 0 && value < (24 * 60)) {
      return Math.round(value);
    }
  }

  const raw = String(value || "").trim();
  if (!raw) return NaN;

  const normalized = raw.match(/^(\d{1,2}):(\d{2})(?::\d{2})?$/);
  if (normalized) {
    return (Number(normalized[1]) * 60) + Number(normalized[2]);
  }

  const formatted = formatTime(value);
  if (formatted && formatted !== raw) {
    const formattedMatch = String(formatted).match(/^(\d{1,2}):(\d{2})$/);
    if (formattedMatch) {
      return (Number(formattedMatch[1]) * 60) + Number(formattedMatch[2]);
    }
  }

  const parsed = new Date(raw);
  if (!isNaN(parsed.getTime())) {
    return (parsed.getHours() * 60) + parsed.getMinutes();
  }

  return NaN;
}

function minutesToTimeText_(minutes) {
  const total = Number(minutes);
  if (!isFinite(total)) return "";
  const h = Math.floor(total / 60);
  const m = total % 60;
  return String(h).padStart(2, "0") + ":" + String(m).padStart(2, "0");
}

function getAppointmentHalfHourSlotCount_(durationMinutes) {
  return Math.max(1, Math.round(normalizeDurationMinutes_(durationMinutes) / 30));
}

function buildHalfHourSlotsFromTime_(startTime, durationMinutes) {
  const startMinutes = timeTextToMinutes_(startTime);
  if (!isFinite(startMinutes)) return [];
  const count = getAppointmentHalfHourSlotCount_(durationMinutes);
  const slots = [];
  for (let i = 0; i < count; i++) {
    slots.push(minutesToTimeText_(startMinutes + (i * 30)));
  }
  return slots;
}

  return raw;
}

function getOccupiedSlotsForDate_(dateString, options) {
  const targetDate = String(dateString || "").trim();
  if (!targetDate) return [];

  const opts = options || {};
  const excludeId = String(opts.exclude_cita_id || opts.appointment_id || "").trim();
  const ensured = ensureAppointmentsSheetHeaders_();
  const headers = ensured.headers;
  const data = ensured.sheet.getDataRange().getValues();
  if (data.length <= 1) return [];

  const idxId = headers.indexOf("id_cita");
  const idxDate = headers.indexOf("fecha");
  const idxTime = headers.indexOf("hora");
  const idxMotivo = headers.indexOf("motivo");
  const idxDuration = headers.indexOf("duracion_minutos");
  if (idxDate === -1 || idxTime === -1) return [];

  const occupied = {};
  for (let i = 1; i < data.length; i++) {
    const rowId = idxId > -1 ? String(data[i][idxId] || "").trim() : "";
    if (excludeId && rowId === excludeId) continue;
    if (toIsoDate(data[i][idxDate]) !== targetDate) continue;

    const duration = resolveAppointmentDurationMinutes_({
      duracion_minutos: idxDuration > -1 ? data[i][idxDuration] : "",
      motivo: idxMotivo > -1 ? data[i][idxMotivo] : ""
    });
    const slots = buildHalfHourSlotsFromTime_(data[i][idxTime], duration);
    slots.forEach(function(slot) {
      occupied[slot] = true;
    });
  }

  return Object.keys(occupied).sort();
}

function getAvailableStartSlotsForDate_(dateString, durationMinutes, options) {
  const occupiedList = getOccupiedSlotsForDate_(dateString, options);
  const occupiedMap = {};
  occupiedList.forEach(function(slot) {
    occupiedMap[slot] = true;
  });

  const dayStart = getAgendaDayStartMinutes_();
  const lastStart = getAgendaLastStartMinutes_();
  const dayEnd = getAgendaDayEndMinutes_();
  const blockMinutes = getAppointmentHalfHourSlotCount_(durationMinutes) * 30;
  const available = [];

  for (let start = dayStart; start <= lastStart; start += 30) {
    const end = start + blockMinutes;
    if (end > dayEnd) continue;

    let free = true;
    for (let cursor = start; cursor < end; cursor += 30) {
      if (occupiedMap[minutesToTimeText_(cursor)]) {
        free = false;
        break;
      }
    }
    if (free) available.push(minutesToTimeText_(start));
  }

  return available;
}

function isAppointmentRangeAvailable_(dateString, startTime, durationMinutes, options) {
  const startMinutes = timeTextToMinutes_(startTime);
  if (!isFinite(startMinutes)) return false;

  const dayStart = getAgendaDayStartMinutes_();
  const dayEnd = getAgendaDayEndMinutes_();
  const blockMinutes = getAppointmentHalfHourSlotCount_(durationMinutes) * 30;
  if (startMinutes < dayStart || (startMinutes + blockMinutes) > dayEnd) return false;

  const occupiedList = getOccupiedSlotsForDate_(dateString, options);
  const occupiedMap = {};
  occupiedList.forEach(function(slot) {
    occupiedMap[slot] = true;
  });

  const neededSlots = buildHalfHourSlotsFromTime_(startTime, durationMinutes);
  for (let i = 0; i < neededSlots.length; i++) {
    if (occupiedMap[neededSlots[i]]) return false;
  }
  return true;
}

// Obtener horas ocupadas (Bloqueo de horario)
function getTakenSlots(dateString, options) {
  const opts = options || {};
  const mode = String(opts.mode || "").trim().toLowerCase();

  if (mode === "available") {
    let duration = parseAllowedDurationMinutes_(opts.duration_minutes);
    if (!duration && opts.appointment_id) {
      duration = getAppointmentDurationMinutesById_(opts.appointment_id);
    }
    return getAvailableStartSlotsForDate_(dateString, duration || 30, {
      exclude_cita_id: opts.exclude_cita_id || opts.appointment_id || ""
    });
  }

  return getOccupiedSlotsForDate_(dateString, opts);
}

function getWeeklyStats() {
  const sheet = getSheet("citas");
  const data = sheet.getDataRange().getValues();
  
  if (data.length <= 1) return { citas_hoy: 0, citas_semana: 0 };
  
  const today = new Date();
  const todayStr = toIsoDate(today); // FIX: Usar triturador

  // Calcular Lunes de esta semana
  const d = new Date(today);
  const day = d.getDay(); 
  const diff = d.getDate() - day + (day == 0 ? -6 : 1); 
  const monday = new Date(d.setDate(diff)); monday.setHours(0,0,0,0);
  
  // Sábado
  const saturday = new Date(monday); saturday.setDate(monday.getDate() + 5); saturday.setHours(23,59,59,999);
  
  let countWeek = 0;
  let countToday = 0;

  for (let i = 1; i < data.length; i++) {
    // FIX: Aplicamos toIsoDate para limpiar lo que viene de Excel
    const rowDateStr = toIsoDate(data[i][2]);
    
    // Contar Hoy
    if (rowDateStr === todayStr) countToday++;
    
    // Contar Semana
    const parts = rowDateStr.split('-');
    if(parts.length === 3) {
       // Crear objeto fecha seguro (Mes base 0)
       const rowDateObj = new Date(parts[0], parts[1]-1, parts[2]);
       rowDateObj.setHours(12,0,0,0); // Poner hora media para evitar bordes
       
       if (rowDateObj >= monday && rowDateObj <= saturday) {
         countWeek++;
       }
    }
  }
  
  return { citas_hoy: countToday, citas_semana: countWeek };
}

function buildWhatsappLink_(phone, fecha, hora) {
  const num = normalizePhone_(phone);
  if (!num) return "";
  const msg = `Gracias por agendar con nosotros te esperamos ${fecha} ${hora}`;
  return "https://wa.me/" + num + "?text=" + encodeURIComponent(msg);
}

function normalizePhone_(phone) {
  if (!phone) return "";
  let digits = String(phone).replace(/[^\d]/g, "");
  if (!digits) return "";
  // Si empieza con 0 y tiene 10 dígitos, asumimos Ecuador (+593)
  if (digits.length === 10 && digits.charAt(0) === "0") {
    digits = "593" + digits.substring(1);
  }
  // Si tiene 9 dígitos, asumimos Ecuador (+593)
  if (digits.length === 9) {
    digits = "593" + digits;
  }
  return digits;
}

function getDoctorContactForPatient_(patientRow) {
  try {
    const createdBy = patientRow ? patientRow.creado_por : "";
    if (!createdBy) {
      return getDefaultDoctorContact_();
    }
    const doctor = findRow("usuarios_admin", "usuario", createdBy);
    if (!doctor) {
      return getDefaultDoctorContact_() || { email: "", name: createdBy };
    }
    const email = doctor.correo_notificaciones || doctor.correo || "";
    const name = doctor.nombre_doctor || doctor.nombre || createdBy;
    const phone = doctor.telefono || "";
    if (!email) {
      const fallback = getDefaultDoctorContact_();
      return fallback || { email: "", name: name, phone: phone };
    }
    return { email: String(email || "").trim(), name: String(name || "").trim(), phone: String(phone || "").trim() };
  } catch (e) {
    return { email: "", name: "" };
  }
}

  const patientPhone = payloadPatientPhone || String((patient && patient.telefono) || "").trim();
  const fecha = String(data.fecha || data.nueva_fecha || "").trim();
  const hora = String(data.hora || data.nueva_hora || "").trim();
  const motivo = String(data.motivo || "").trim();
  const recomendaciones = String(data.recomendaciones || "").trim();
  const doctorInfo = {
    email: payloadDoctorEmail,
    name: payloadDoctorName,
    phone: payloadDoctorPhone
  };
  if ((!doctorInfo.email || !doctorInfo.name || !doctorInfo.phone) && patient) {
    const sheetDoctor = getDoctorContactForPatient_(patient) || { email: "", name: "", phone: "" };
    if (!doctorInfo.email) doctorInfo.email = String(sheetDoctor.email || "").trim();
    if (!doctorInfo.name) doctorInfo.name = String(sheetDoctor.name || "").trim();
    if (!doctorInfo.phone) doctorInfo.phone = String(sheetDoctor.phone || "").trim();
  }
  if ((!doctorInfo.email || !doctorInfo.name || !doctorInfo.phone) && !patient) {
    const fallbackDoctor = getDefaultDoctorContact_() || { email: "", name: "", phone: "" };
    if (!doctorInfo.email) doctorInfo.email = String(fallbackDoctor.email || "").trim();
    if (!doctorInfo.name) doctorInfo.name = String(fallbackDoctor.name || "").trim();
    if (!doctorInfo.phone) doctorInfo.phone = String(fallbackDoctor.phone || "").trim();
  }

  let sentPatient = false;
  let sentDoctor = false;
  const doctorPhone = String(doctorInfo.phone || "").trim();

  try {
    if (eventType === "schedule") {
      if (patientEmail && patientEmail.indexOf("@") !== -1) {
        sendAppointmentEmailToPatient(patientEmail, patientName, {
          fecha: fecha,
          hora: hora,
          recomendaciones: recomendaciones
        }, appointmentId);
        sentPatient = true;
      }
      if (doctorInfo.email && String(doctorInfo.email).indexOf("@") !== -1) {
        sendAppointmentEmailToDoctor(
          doctorInfo.email,
          doctorInfo.name,
          patientName,
          patientPhone,
          { fecha: fecha, hora: hora, motivo: motivo, recomendaciones: recomendaciones },
          appointmentId,
          false
        );
        sentDoctor = true;
      }
    } else if (eventType === "reschedule") {
      sentPatient = sendRescheduleEmailToPatient_(patientEmail, patientName, fecha, hora) || false;
      if (doctorInfo.email && String(doctorInfo.email).indexOf("@") !== -1) {
        sendAppointmentEmailToDoctor(
          doctorInfo.email,
          doctorInfo.name,
          patientName,
          patientPhone,
          { fecha: fecha, hora: hora, motivo: motivo || "REAGENDADO" },
          appointmentId,
          true
        );
        sentDoctor = true;
      }
    } else if (eventType === "cancel" || eventType === "delete") {
      sentPatient = sendAppointmentCancellationEmailToPatient_(patientEmail, patientName, fecha, hora, motivo) || false;
      if (doctorInfo.email && String(doctorInfo.email).indexOf("@") !== -1) {
        sendAppointmentCancellationEmailToDoctor_(
          doctorInfo.email,
          doctorInfo.name,
          patientName,
          patientPhone,
          { fecha: fecha, hora: hora, motivo: motivo || "Cita cancelada" },
          appointmentId
        );
        sentDoctor = true;
      }
    } else {
      return { success: false, message: "Evento de notificacion no soportado." };
    }
  } catch (e) {
    return { success: false, message: "Error enviando notificaciones: " + e.toString() };
  }

  if (!sentPatient && !sentDoctor) {
    return {
      success: false,
      message: "No se envio ningun correo. Verifica el correo del paciente y el correo_notificaciones del doctor.",
      sent_patient_email: false,
      sent_doctor_email: false,
      doctor_phone: doctorPhone
    };
  }

  return {
    success: true,
    message: "Notificaciones procesadas.",
    sent_patient_email: !!sentPatient,
    sent_doctor_email: !!sentDoctor,
    doctor_phone: doctorPhone
  };
}

function rescheduleAppointment(dataObj) {
  const ensured = ensureAppointmentsSheetHeaders_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const rowIndex = findRowIndex(sheet, "id_cita", dataObj.id_cita);
  
  if (rowIndex === -1) return { success: false, message: "Cita no encontrada." };

  const idxReason = headers.indexOf("motivo");
  const idxDuration = headers.indexOf("duracion_minutos");
  const currentRow = sheet.getRange(rowIndex, 1, 1, headers.length).getValues()[0];
  const durationMinutes = resolveAppointmentDurationMinutes_({
    duracion_minutos: idxDuration > -1 ? currentRow[idxDuration] : "",
    motivo: idxReason > -1 ? currentRow[idxReason] : ""
  });
  if (!isAppointmentRangeAvailable_(dataObj.nueva_fecha, dataObj.nueva_hora, durationMinutes, { exclude_cita_id: dataObj.id_cita })) {
    return { success: false, message: "Ese horario ya no esta disponible para la duracion de esta cita." };
  }
  
  // Actualizamos Fecha (Col 3 / Índice 2) y Hora (Col 4 / Índice 3)
  // Nota: Índices de columna en getRange son base 1. Fecha=3, Hora=4.
  // Revisa tu estructura: id(1), id_pac(2), fecha(3), hora(4), motivo(5), estado(6)...
  
  // 1. Actualizar en Excel
  const idxDate = headers.indexOf("fecha");
  const idxTime = headers.indexOf("hora");
  const idxState = headers.indexOf("estado");
  if (idxDate > -1) sheet.getRange(rowIndex, idxDate + 1).setValue(dataObj.nueva_fecha);
  if (idxTime > -1) sheet.getRange(rowIndex, idxTime + 1).setValue(dataObj.nueva_hora);
  if (idxState > -1) sheet.getRange(rowIndex, idxState + 1).setValue("REAGENDADO");
  
  // 2. Buscar datos para el correo
  const idPaciente = sheet.getRange(rowIndex, 2).getValue();
  const paciente = findRow("pacientes", "id_paciente", idPaciente);
  
  // 3. Enviar correo al paciente
  if (paciente) sendRescheduleEmailToPatient_(paciente.correo, paciente.nombre_completo, dataObj.nueva_fecha, dataObj.nueva_hora);

  // 4. Notificar al médico (si existe) con botón WhatsApp al paciente
  let doctorInfo = { email: "", name: "", phone: "" };
  try {
    doctorInfo = getDoctorContactForPatient_(paciente) || { email: "", name: "", phone: "" };
    sendAppointmentEmailToDoctor(doctorInfo.email, doctorInfo.name, paciente.nombre_completo, paciente.telefono, { 
      fecha: dataObj.nueva_fecha, 
      hora: dataObj.nueva_hora, 
      motivo: "REAGENDADO"
    }, dataObj.id_cita, true);
  } catch(e) {}

  const response = { success: true, message: "Cita reagendada correctamente.", doctor_phone: doctorInfo.phone || "" };
  const syncRes = syncAppointmentToSupabase_(findRow("citas", "id_cita", dataObj.id_cita) || {
    id_cita: dataObj.id_cita,
    id_paciente: idPaciente,
    fecha: dataObj.nueva_fecha,
    hora: dataObj.nueva_hora,
    estado: "REAGENDADO"
  });
  if (syncRes && !syncRes.success && !syncRes.skipped) {
    response.warning = syncRes.message || "No se pudo sincronizar con Supabase.";
    response.message = "Cita reagendada correctamente. Advertencia: no se pudo sincronizar con Supabase.";
  }
  return response;
}

function ensureClinicalHistorySheet_() {
  const sheet = getSheet("historia_clinica");
  const required = [
    "id_paciente", "app", "apf", "alergias", "aqx",
    "menarquia", "prs", "num_parejas",
    "ago_g", "ago_p", "ago_c", "ago_a",
    "fecha_aborto", "pap", "fum",
    "anticonceptivos", "tipo_anti", "tiempo_uso",
    "tipo_ultimo", "fecha_actualizacion"
  ];

  if (sheet.getLastRow() === 0) {
    sheet.appendRow(required);
    return { sheet: sheet, headers: required };
  }

  const lastCol = Math.max(sheet.getLastColumn(), required.length);
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  required.forEach(function(name) {
    if (headers.indexOf(name) !== -1) return;
    const col = sheet.getLastColumn() + 1;
    sheet.getRange(1, col).setValue(name);
    headers.push(name);
    if (sheet.getLastRow() > 1) {
      sheet.getRange(2, col, sheet.getLastRow() - 1, 1).setValues(
        new Array(sheet.getLastRow() - 1).fill([""])
      );
    }
  });

  return { sheet: sheet, headers: headers };
}

function buildClinicalHistoryFromSheetRow_(headers, row) {
  const rowObj = mapSheetRowObject_(headers, row);
  const tipoUlt = String(rowObj.tipo_ultimo || "").trim()
    || (String(rowObj.es_parto || "").trim().toUpperCase() === "SI" ? "Parto" : "")
    || (String(rowObj.es_aborto || "").trim().toUpperCase() === "SI" ? "Aborto" : "");

  return {
    id_paciente: String(rowObj.id_paciente || "").trim(),
    app: String(rowObj.app || "").trim(),
    apf: String(rowObj.apf || "").trim(),
    alergias: String(rowObj.alergias || "").trim(),
    aqx: String(rowObj.aqx || "").trim(),
    menarquia: String(rowObj.menarquia || "").trim(),
    prs: String(rowObj.prs || "").trim(),
    num_parejas: String(rowObj.num_parejas || "").trim(),
    ago_g: String(rowObj.ago_g || "").trim(),
    ago_p: String(rowObj.ago_p || "").trim(),
    ago_c: String(rowObj.ago_c || "").trim(),
    ago_a: String(rowObj.ago_a || "").trim(),
    fecha_aborto: toIsoDate(rowObj.fecha_aborto || rowObj.fecha_ultimo_evento),
    pap: String(rowObj.pap || "").trim(),
    fum: toIsoDate(rowObj.fum),
    anticonceptivos: String(rowObj.anticonceptivos || "").trim(),
    tipo_anti: String(rowObj.tipo_anti || "").trim(),
    tiempo_uso: String(rowObj.tiempo_uso || "").trim(),
    tipo_ultimo: tipoUlt,
    ante_its: String(rowObj.ante_its || "").trim(),
    tipo_its: String(rowObj.tipo_its || "").trim(),
    fecha_actualizacion: rowObj.fecha_actualizacion || ""
  };
}

function buildClinicalHistorySheetRow_(headers, dataObj, updatedAt) {
  const row = new Array(headers.length).fill("");
  const source = dataObj || {};
  const tipoUlt = String(source.tipo_ultimo || "").trim();
  const fechaAborto = String(source.fecha_aborto || source.fecha_ultimo_evento || "").trim();

  for (let i = 0; i < headers.length; i++) {
    const key = headers[i];
    if (key === "fecha_actualizacion") {
      row[i] = updatedAt || new Date();
    } else if (key === "fecha_aborto" || key === "fecha_ultimo_evento") {
      row[i] = fechaAborto;
    } else if (key === "tipo_ultimo") {
      row[i] = tipoUlt;
    } else if (key === "es_parto") {
      row[i] = tipoUlt.toUpperCase() === "PARTO" ? "SI" : "NO";
    } else if (key === "es_aborto") {
      row[i] = tipoUlt.toUpperCase() === "ABORTO" ? "SI" : "NO";
    } else if (Object.prototype.hasOwnProperty.call(source, key)) {
      row[i] = source[key];
    }
  }
  return row;
}

function getClinicalHistory(patientId) {
  const ensured = ensureClinicalHistorySheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return {};

  for (let i = 1; i < data.length; i++) {
    const mapped = buildClinicalHistoryFromSheetRow_(headers, data[i]);
    if (String(mapped.id_paciente) === String(patientId)) {
      return mapped;
    }
  }
  return {};
}

function saveClinicalHistory(dataObj) {
  const ensured = ensureClinicalHistorySheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const patientId = String((dataObj && dataObj.id_paciente) || "").trim();
  if (!patientId) {
    return { success: false, message: "Falta id_paciente." };
  }

  const rowIndex = findRowIndex(sheet, "id_paciente", patientId);
  const updatedAt = new Date();
  const rowData = buildClinicalHistorySheetRow_(headers, dataObj || {}, updatedAt);

  if (rowIndex !== -1) {
    const range = sheet.getRange(rowIndex, 1, 1, headers.length);
    range.setNumberFormat('@');
    range.setValues([rowData]);
  } else {
    sheet.appendRow(rowData);
  }

  const historyRecord = getClinicalHistory(patientId);
  if (!historyRecord.fecha_actualizacion) {
    historyRecord.fecha_actualizacion = updatedAt;
  }

  const response = { success: true, message: "Historia clínica guardada." };
  const syncRes = syncClinicalHistoryToSupabase_(historyRecord);
  if (syncRes && !syncRes.success && !syncRes.skipped) {
    response.warning = syncRes.message || "No se pudo sincronizar con Supabase.";
    response.message = "Historia clínica guardada. Advertencia: no se pudo sincronizar con Supabase.";
    try {
      var warningPayload = {
        patient_id: patientId,
        sync_error: response.warning
      };
      console.log("[HISTORIA_SYNC_WARNING] " + JSON.stringify(warningPayload));
      Logger.log("[HISTORIA_SYNC_WARNING] " + JSON.stringify(warningPayload));
    } catch (e) {}
  }
  return response;
}

function normalizePromotionLower_(value) {
  return String(value === undefined || value === null ? "" : value).trim().toLowerCase();
}

function normalizePromotionScope_(value) {
  return String(value || "").trim().toUpperCase() === "ALL" ? "ALL" : "OWNER";
}

function ensurePromotionSheet_() {
  const sheet = getSheet("config_promociones");
  const required = [
    "id_promo",
    "mensaje",
    "fecha_inicio",
    "fecha_fin",
    "scope_visibility",
    "owner_usuario",
    "fecha_creacion"
  ];

  if (sheet.getLastRow() === 0) {
    sheet.appendRow(required);
    return { sheet: sheet, headers: required };
  }

  const lastCol = Math.max(sheet.getLastColumn(), required.length);
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  required.forEach(function(name) {
    if (headers.indexOf(name) !== -1) return;
    const col = sheet.getLastColumn() + 1;
    sheet.getRange(1, col).setValue(name);
    headers.push(name);
  });

  return { sheet: sheet, headers: headers };
}

function mapPromotionRow_(headers, row, rowIndex) {
  const idxId = headers.indexOf("id_promo");
  const idxMsg = headers.indexOf("mensaje");
  const idxStart = headers.indexOf("fecha_inicio");
  const idxEnd = headers.indexOf("fecha_fin");
  const idxScope = headers.indexOf("scope_visibility");
  const idxOwner = headers.indexOf("owner_usuario");
  const idxCreated = headers.indexOf("fecha_creacion");

  const id = idxId > -1 ? String(row[idxId] || "").trim() : "";
  if (!id) return null;
  return {
    rowIndex: rowIndex,
    id: id,
    mensaje: idxMsg > -1 ? String(row[idxMsg] || "").trim() : "",
    inicio: idxStart > -1 ? toIsoDate(row[idxStart]) : "",
    fin: idxEnd > -1 ? toIsoDate(row[idxEnd]) : "",
    scope_visibility: normalizePromotionScope_(idxScope > -1 ? row[idxScope] : ""),
    owner_usuario: normalizePromotionLower_(idxOwner > -1 ? row[idxOwner] : ""),
    fecha_creacion: idxCreated > -1 ? toIsoDate(row[idxCreated]) : ""
  };
}

function getPromotionViewerDoctor_(requester) {
  const req = String(requester || "").trim();
  if (!req) return "";
  const patient = findRow("pacientes", "id_paciente", req);
  if (patient) return normalizePromotionLower_(patient.creado_por || "");
  return normalizePromotionLower_(req);
}

function savePromotion(dataObj, requester) {
  const data = dataObj || {};
  const message = String(data.mensaje || "").trim();
  const inicio = String(data.fecha_inicio || "").trim();
  const fin = String(data.fecha_fin || "").trim();
  const scope = normalizePromotionScope_(data.scope_visibility);
  const owner = normalizePromotionLower_(requester);

  if (!message || !inicio || !fin) {
    return { success: false, message: "Completa mensaje, fecha_inicio y fecha_fin." };
  }

  const ensured = ensurePromotionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const idPromo = "PROMO-" + new Date().getTime();

  const row = new Array(headers.length).fill("");
  const idxId = headers.indexOf("id_promo");
  const idxMsg = headers.indexOf("mensaje");
  const idxStart = headers.indexOf("fecha_inicio");
  const idxEnd = headers.indexOf("fecha_fin");
  const idxScope = headers.indexOf("scope_visibility");
  const idxOwner = headers.indexOf("owner_usuario");
  const idxCreated = headers.indexOf("fecha_creacion");

  if (idxId > -1) row[idxId] = idPromo;
  if (idxMsg > -1) row[idxMsg] = message;
  if (idxStart > -1) row[idxStart] = inicio;
  if (idxEnd > -1) row[idxEnd] = fin;
  if (idxScope > -1) row[idxScope] = scope;
  if (idxOwner > -1) row[idxOwner] = owner;
  if (idxCreated > -1) row[idxCreated] = new Date();

  sheet.appendRow(row);
  const saved = mapPromotionRow_(headers, row, sheet.getLastRow());
  const syncRes = syncPromotionToSupabase_(saved);
  if (syncRes && !syncRes.success && !syncRes.skipped) {
    return {
      success: true,
      message: "Promocion guardada. Advertencia: no se pudo sincronizar con Supabase.",
      warning: syncRes.message || "No se pudo sincronizar con Supabase."
    };
  }
  return { success: true, message: "Promocion guardada." };
}

function getActivePromotion(requester) {
  const ensured = ensurePromotionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return { success: true, active: false };

  const viewerDoctor = getPromotionViewerDoctor_(requester);
  const today = Utilities.formatDate(new Date(), "GMT-5", "yyyy-MM-dd");

  for (let i = data.length - 1; i >= 1; i--) {
    const mapped = mapPromotionRow_(headers, data[i], i + 1);
    if (!mapped || !mapped.inicio || !mapped.fin) continue;
    if (!(today >= mapped.inicio && today <= mapped.fin)) continue;

    if (mapped.scope_visibility === "ALL") {
      return {
        success: true,
        active: true,
        id: mapped.id,
        mensaje: mapped.mensaje,
        fin: mapped.fin,
        scope_visibility: mapped.scope_visibility
      };
    }

    if (mapped.owner_usuario && viewerDoctor && mapped.owner_usuario === viewerDoctor) {
      return {
        success: true,
        active: true,
        id: mapped.id,
        mensaje: mapped.mensaje,
        fin: mapped.fin,
        scope_visibility: mapped.scope_visibility
      };
    }
  }

  return { success: true, active: false };
}

function getPromoConfig(requester) {
  const ensured = ensurePromotionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return { success: true, list: [] };

  const req = normalizePromotionLower_(requester);
  const isSuper = !!(typeof isSuperadminUser === "function" && req && isSuperadminUser(req));
  const list = [];

  for (let i = 1; i < data.length; i++) {
    const mapped = mapPromotionRow_(headers, data[i], i + 1);
    if (!mapped) continue;
    if (!isSuper && req && mapped.owner_usuario && mapped.owner_usuario !== req) continue;
    list.push({
      id: mapped.id,
      mensaje: mapped.mensaje,
      inicio: mapped.inicio,
      fin: mapped.fin,
      scope_visibility: mapped.scope_visibility,
      owner_usuario: mapped.owner_usuario
    });
  }

  return { success: true, list: list.reverse() };
}

function deletePromotion(idPromo, requester) {
  const targetId = String(idPromo || "").trim();
  const req = normalizePromotionLower_(requester);
  if (!targetId || !req) return { success: false, message: "Datos incompletos." };

  const ensured = ensurePromotionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return { success: false, message: "Promocion no encontrada." };

  const isSuper = !!(typeof isSuperadminUser === "function" && isSuperadminUser(req));
  let target = null;
  for (let i = 1; i < data.length; i++) {
    const mapped = mapPromotionRow_(headers, data[i], i + 1);
    if (!mapped || mapped.id !== targetId) continue;
    target = mapped;
    break;
  }

  if (!target) return { success: false, message: "Promocion no encontrada." };
  if (!isSuper && target.owner_usuario && target.owner_usuario !== req) {
    return { success: false, message: "Solo el medico creador puede eliminar esta promocion." };
  }

  sheet.deleteRow(target.rowIndex);
  const syncRes = deletePromotionFromSupabase_(target.id);
  if (syncRes && !syncRes.success && !syncRes.skipped) {
    return {
      success: true,
      message: "Promocion eliminada. Advertencia: no se pudo sincronizar con Supabase.",
      warning: syncRes.message || "No se pudo sincronizar con Supabase."
    };
  }
  return { success: true, message: "Promocion eliminada." };
}

              const file = reportFolder.createFile(blob); 
              finalFileIds.push(file.getId());
              imgsForDoc[img.index] = blob; 
              img.fileId = file.getId();
              img.data = null; 
            } else if (img.fileId) {
                 finalFileIds.push(img.fileId);
                 try { imgsForDoc[img.index] = DriveApp.getFileById(img.fileId).getBlob(); } catch(e){}
            }
        } catch(e) { console.log("Error foto: " + e); }
      });
    }
    if (!finalFileIds.length && isUpdate && oldJsonData.drive_file_ids && Array.isArray(oldJsonData.drive_file_ids)) {
      oldJsonData.drive_file_ids.forEach(function(fileId) {
        const cleanId = String(fileId || "").trim();
        if (!cleanId || finalFileIds.indexOf(cleanId) !== -1) return;
        finalFileIds.push(cleanId);
      });
    }

    // 4. PDF EXTERNO
    let pdfExternos = handleExternalPdf(dataObj, isUpdate, oldJsonData, reportFolder);
    let pdfExternoUrl = pdfExternos.length ? String(pdfExternos[0].url || "").trim() : "";

    // 5. GENERAR PDFS (Reporte y Receta)
    const examType = String(dataObj.tipo_examen || "").trim().toUpperCase();
    const isRecipeOnly = examType === "RECETA";
    const isExternalPdfOnly = examType === "EXAMENPDF";
    const hasClinicalReportContent = diagnosisHasMeaningfulReportContent_(dataObj);
    const skipClinicalReportPdf = isRecipeOnly || isExternalPdfOnly || !hasClinicalReportContent;
    if (isExternalPdfOnly && !pdfExternoUrl) {
        return { success: false, message: "Debes adjuntar un PDF para guardar en EXAMENPDF." };
    }
    let pdfUrl = oldPdfUrl || "";
    let pdfRecetaUrl = oldJsonData.pdf_receta_link || "";

    if (dataObj.generar_pdf) {
        // A. REPORTE
        if (!skipClinicalReportPdf) {
            try {
                if (isUpdate && oldPdfUrl && oldPdfUrl.includes("id=")) {
                    try { DriveApp.getFileById(oldPdfUrl.match(/id=([a-zA-Z0-9_-]+)/)[1]).setTrashed(true); } catch(e) {}
                }
                pdfUrl = createPdfFromTemplateCompact(dataObj, imgsForDoc, reportFolder);
            } catch (e) { console.log("Error PDF: " + e.message); }
        }
        // B. RECETA
        if (dataObj.medicamentos && dataObj.medicamentos.length > 0) {
            try {
                if (isUpdate && oldJsonData.pdf_receta_link) {
                    try { DriveApp.getFileById(oldJsonData.pdf_receta_link.match(/id=([a-zA-Z0-9_-]+)/)[1]).setTrashed(true); } catch(e){}
                }
                pdfRecetaUrl = createRecetaDocFixed(dataObj, reportFolder); 
            } catch (e) { console.log("Error Receta: " + e.message); }
        }
    }

    // 6. GUARDAR EN EXCEL
    const savedAt = new Date();
    const mergedDataObj = mergeDiagnosisPayloadForSave_(oldJsonData, dataObj, examType);
    saveToExcel(sheet, rowIndex, isUpdate, idReporte, mergedDataObj, finalFileIds, reportFolder.getId(), pdfUrl, pdfRecetaUrl, pdfExternos, savedAt);

    const cleanSync = { ...mergedDataObj };
    cleanSync.drive_file_ids = finalFileIds;
    cleanSync.report_folder_id = reportFolder.getId();
    cleanSync.pdf_receta_link = pdfRecetaUrl;
    cleanSync.pdf_externos = pdfExternos;
    cleanSync.pdf_externo_link = pdfExternoUrl;
    delete cleanSync.pdf_externo;
    if (cleanSync.imagenes) {
      cleanSync.imagenes = cleanSync.imagenes.map(function(img) {
        return { index: img.index, title: img.title || "", fileId: img.fileId, isNew: false };
      });
    }
    const datosJsonSync = JSON.stringify(cleanSync);
    const diagnosisRow = {
      id_reporte: idReporte,
      id_paciente: dataObj.id_paciente,
      tipo_examen: dataObj.tipo_examen,
      fecha: savedAt,
      datos_json: datosJsonSync,
      pdf_url: pdfUrl,
      creado_por: oldCreatedBy || "DOCTOR"
    };
    const syncRes = syncDiagnosisToSupabase_(diagnosisRow);
    if (syncRes && !syncRes.success && !syncRes.skipped) {
      return {
        success: true,
        message: "Guardado exitoso. Advertencia: no se pudo sincronizar con Supabase.",
        warning: syncRes.message || "No se pudo sincronizar con Supabase.",
        id_reporte: idReporte,
        pdf_url: pdfUrl,
        pdf_receta_url: pdfRecetaUrl,
        pdf_externo_url: pdfExternoUrl,
        pdf_externos: pdfExternos
      };
    }

    return { 
        success: true, 
        message: "Guardado exitoso", 
        id_reporte: idReporte, // IMPORTANTE: Devolvemos el ID
        pdf_url: pdfUrl,
        pdf_receta_url: pdfRecetaUrl,
        pdf_externo_url: pdfExternoUrl,
        pdf_externos: pdfExternos
    };

  } catch (error) {
    return { success: false, message: "ERROR CRÍTICO: " + error.toString() };
  } finally {
    lock.releaseLock();
  }
}

function parseVirtualSignatureBool_(value, fallback) {
  const defaultValue = fallback === undefined ? true : !!fallback;
  if (value === undefined || value === null || value === "") return defaultValue;
  if (value === true || value === 1) return true;
  if (value === false || value === 0) return false;
  const raw = String(value || "").trim().toLowerCase();
  if (!raw) return defaultValue;
  if (raw === "si" || raw === "true" || raw === "1" || raw === "yes" || raw === "on") return true;
  if (raw === "no" || raw === "false" || raw === "0" || raw === "off") return false;
  return defaultValue;
}

function ensureDoctorSignatureColumnsInAdmin_() {
  const sheet = getConnection().getSheetByName("usuarios_admin");
  if (!sheet || sheet.getLastRow() === 0) return;

  const required = [
    { name: "registro_sanitario", defaultValue: "" },
    { name: "usar_firma_virtual", defaultValue: "SI" }
  ];

  const headers = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), required.length)).getValues()[0];
  required.forEach(function(def) {
    if (headers.indexOf(def.name) !== -1) return;
    const col = sheet.getLastColumn() + 1;
    sheet.getRange(1, col).setValue(def.name);
    headers.push(def.name);
    if (sheet.getLastRow() > 1) {
      sheet.getRange(2, col, sheet.getLastRow() - 1, 1).setValues(
        new Array(sheet.getLastRow() - 1).fill([def.defaultValue])
      );
    }
  });
}

function getDoctorVirtualSignatureMeta_(doctorUser) {
  const user = String(doctorUser || "").trim().toLowerCase();
  if (!user) {
    return {
      usuario: "",
      nombre_doctor: "",
      rol: "DOCTOR",
      registro_sanitario: "",
      usar_firma_virtual: true
    };
  }

  const doctor = findRow("usuarios_admin", "usuario", user) || {};

  return {
    usuario: user,
    nombre_doctor: String(doctor.nombre_doctor || doctor.nombre || user).trim(),
    rol: String(doctor.rol || doctor.ocupacion || "DOCTOR").trim(),
    registro_sanitario: String(doctor.registro_sanitario || "").trim(),
    usar_firma_virtual: parseVirtualSignatureBool_(doctor.usar_firma_virtual, true)
  };
}

function shouldIncludeDoctorVirtualSignature_(data, meta) {
  if (data && Object.prototype.hasOwnProperty.call(data, "incluir_firma_virtual")) {
    return parseVirtualSignatureBool_(data.incluir_firma_virtual, meta ? meta.usar_firma_virtual : true);
  }
  return meta ? !!meta.usar_firma_virtual : true;
}

function formatDoctorRoleForVirtualSignature_(role) {
  const text = String(role || "DOCTOR").trim().toUpperCase();
  if (!text) return "DOCTOR";
  if (text.length > 22 && text.indexOf(" Y ") > -1) {
    return text.replace(" Y ", "\nY ");
  }
  if (text.length > 24 && text.indexOf(" / ") > -1) {
    return text.replace(" / ", "\n");
  }
  return text;
}

function escapeVirtualSignatureSvgText_(value) {
  return String(value || "")
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");
}

function buildDoctorVirtualSignatureSealSvgBlob_(meta) {
  const doctorName = escapeVirtualSignatureSvgText_(String(meta && meta.nombre_doctor || "").trim());
  const roleText = formatDoctorRoleForVirtualSignature_(meta && meta.rol || "DOCTOR");
  const roleLines = String(roleText || "DOCTOR").split("\n").filter(function(line) {
    return String(line || "").trim();
  });
  const safeRoleLines = roleLines.length ? roleLines.slice(0, 3) : ["DOCTOR"];
  const registerLine = String(meta && meta.registro_sanitario || "").trim()
    ? "Reg. San. " + String(meta.registro_sanitario || "").trim()
    : "";

  const svgParts = [
    '<svg xmlns="http://www.w3.org/2000/svg" width="550" height="146" viewBox="0 0 550 146">',
    '<rect width="550" height="146" fill="transparent"/>',
    '<g transform="translate(92 20)">',
    '<g transform="translate(0 8) scale(0.28515625)">',
    '<circle cx="128" cy="128" r="120" fill="#43286d"/>',
    '<circle cx="141" cy="67" r="30" fill="#c9c8e8"/>',
    '<circle cx="101" cy="122" r="22" fill="#b6b3da"/>',
    '<path d="M123 98 C153 91 176 108 178 139 C180 178 157 210 124 237 L124 152 C124 129 123 113 123 98 Z" fill="#b79ac8"/>',
    '<path d="M68 116 C59 140 66 170 88 194 C98 205 111 217 124 230 L124 191 C101 178 82 151 77 124 C74 120 72 118 68 116 Z" fill="#d7b7dd"/>',
    '</g>'
  ];

  const nameStartX = 0;
  const roleStartY = safeRoleLines.length > 1 ? 45 : 52;
  const roleStep = 14;
  svgParts.push(
    '<text x="' + String(nameStartX) + '" y="18" text-anchor="start" fill="#2f2541" font-family="Georgia, Times New Roman, serif" font-size="17" font-style="italic" font-weight="700">',
    doctorName,
    '</text>'
  );
  svgParts.push('<g transform="translate(82 0)">');
  for (var i = 0; i < safeRoleLines.length; i++) {
    svgParts.push(
      '<text x="0" y="' + String(roleStartY + (i * roleStep)) + '" text-anchor="start" fill="#4a386d" font-family="Arial, Helvetica, sans-serif" font-size="13" font-weight="700">',
      escapeVirtualSignatureSvgText_(safeRoleLines[i]),
      '</text>'
    );
  }
  if (registerLine) {
    svgParts.push(
      '<text x="0" y="' + String(roleStartY + (safeRoleLines.length * roleStep) + 7) + '" text-anchor="start" fill="#5a5a5a" font-family="Arial, Helvetica, sans-serif" font-size="11" font-style="italic" font-weight="600">',
      escapeVirtualSignatureSvgText_(registerLine),
      '</text>'
    );
  }
  svgParts.push('</g>');
  svgParts.push('</g>');
  svgParts.push('</svg>');

  return Utilities.newBlob(svgParts.join(""), "image/svg+xml", "vidafem_virtual_signature_seal.svg");
}

function appendDoctorVirtualSignature_(body, data) {
  if (!body || !data) return false;

  const meta = getDoctorVirtualSignatureMeta_(data.doctor_usuario || data.creado_por || "");
  if (!shouldIncludeDoctorVirtualSignature_(data, meta)) return false;

  const doctorName = String(meta.nombre_doctor || meta.usuario || "").trim();
  if (!doctorName) return false;

  const doctorRole = formatDoctorRoleForVirtualSignature_(meta.rol);
  const registerLine = String(meta.registro_sanitario || "").trim()
    ? "Reg. San. " + String(meta.registro_sanitario || "").trim()
    : "";

  try {
    const sealParagraph = body.appendParagraph("");
    sealParagraph
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
      .setSpacingBefore(4)
      .setSpacingAfter(0);
    const sealImage = sealParagraph.appendInlineImage(buildDoctorVirtualSignatureSealSvgBlob_(meta));
    sealImage.setWidth(282).setHeight(75);
    return true;
  } catch (eSeal) {}

  body.appendParagraph("").setSpacingBefore(4).setSpacingAfter(0);

  const outer = body.appendTable();
  outer.setBorderWidth(0);
  const outerRow = outer.appendTableRow();
  outerRow.appendTableCell("");
  const centerCell = outerRow.appendTableCell("");
  outerRow.appendTableCell("");
  outer.setColumnWidth(0, 124);
  outer.setColumnWidth(1, 252);
  outer.setColumnWidth(2, 124);

  centerCell.setPaddingTop(0).setPaddingBottom(0).setPaddingLeft(0).setPaddingRight(0);

  const pName = centerCell.getChild(0).asParagraph();
  pName.setText(doctorName);
  pName
    .setAlignment(DocumentApp.HorizontalAlignment.LEFT)
    .setBold(true)
    .setItalic(true)
    .setFontSize(12)
    .setForegroundColor("#2f2541")
    .setIndentStart(0)
    .setSpacingBefore(0)
    .setSpacingAfter(0);

  const inner = centerCell.appendTable();
  inner.setBorderWidth(0);
  const innerRow = inner.appendTableRow();
  const logoCell = innerRow.appendTableCell("");
  const textCell = innerRow.appendTableCell("");
  inner.setColumnWidth(0, 54);
  inner.setColumnWidth(1, 198);

  logoCell
    .setPaddingTop(0)
    .setPaddingBottom(0)
    .setPaddingLeft(2)
    .setPaddingRight(0)
    .setVerticalAlignment(DocumentApp.VerticalAlignment.MIDDLE);
  textCell
    .setPaddingTop(0)
    .setPaddingBottom(0)
    .setPaddingLeft(0)
    .setPaddingRight(0)
    .setVerticalAlignment(DocumentApp.VerticalAlignment.MIDDLE);

  try {
    const logoParagraph = logoCell.getChild(0).asParagraph();
    logoParagraph
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
      .setSpacingBefore(0)
      .setSpacingAfter(0);
    let logoImg = null;
    try {
      logoImg = logoParagraph.appendInlineImage(getVirtualSignatureLogoSvgBlob_());
    } catch (eSvg) {
      logoImg = logoParagraph.appendInlineImage(getVirtualSignatureLogoPngFallbackBlob_());
    }
    if (logoImg) logoImg.setWidth(46).setHeight(46);
  } catch (e) {}

  const roleLines = String(doctorRole || "").split("\n").filter(function(line) {
    return String(line || "").trim();
  });
  const firstRoleLine = roleLines.length ? roleLines.shift() : "DOCTOR";

  const pRole = textCell.getChild(0).asParagraph();
  pRole.setText(firstRoleLine);
  pRole
    .setAlignment(DocumentApp.HorizontalAlignment.LEFT)
    .setBold(true)
    .setFontSize(9)
    .setForegroundColor("#4a386d")
    .setSpacingAfter(0)
    .setSpacingBefore(0);

  for (let i = 0; i < roleLines.length; i++) {
    textCell.appendParagraph(roleLines[i])
      .setAlignment(DocumentApp.HorizontalAlignment.LEFT)
      .setBold(true)
      .setFontSize(9)
      .setForegroundColor("#4a386d")
      .setSpacingAfter(0)
      .setSpacingBefore(0);
  }

  if (registerLine) {
    textCell.appendParagraph(registerLine)
      .setAlignment(DocumentApp.HorizontalAlignment.LEFT)
      .setBold(false)
      .setItalic(true)
      .setFontSize(8)
      .setForegroundColor("#5a5a5a")
      .setSpacingBefore(0)
      .setSpacingAfter(0);
  }

  return true;
}

function createPdfFromTemplateCompact(data, imgsMap, folder) {
  function normalizeTextSafe(s) {
    return String(s || "")
      .replace(/\u00A0/g, " ")
      .normalize("NFD")
      .replace(/[\u0300-\u036f]/g, "")
      .trim()
      .toLowerCase();
  }

  function getBodyTextSafe(b) {
    try { return normalizeTextSafe(b.getText()); } catch (e) { return ""; }
  }

  function getElementTextSafe(el) {
    try {
      if (el.getType() === DocumentApp.ElementType.PARAGRAPH) return normalizeTextSafe(el.asParagraph().getText());
      if (el.getType() === DocumentApp.ElementType.LIST_ITEM) return normalizeTextSafe(el.asListItem().getText());
    } catch (e) {}
    return "";
  }

  function cleanupTextArtifactsInSection(section, normalizedDocName) {
    if (!section) return;
    try {
      section.replaceText("(i)\\s*(pestaña|pestana|tab|document\\s*tab)\\s*\\d+\\s*", "");
      section.replaceText("(i)\\s*(pestaña|pestana)\\s*sin\\s*titulo\\s*", "");
      section.replaceText("(i)\\s*untitled\\s*tab\\s*", "");
      section.replaceText("(i)\\s*tab\\s*without\\s*title\\s*", "");
      if (normalizedDocName) {
        section.replaceText("(i)\\s*" + normalizedDocName.replace(/[.*+^${}()|[\]\\]/g, "\\$&") + "\\s*", "");
      }
    } catch (e) {}
  }

  function cleanupDocTabs(doc, expectedDocName) {
    if (typeof doc.getTabs !== "function") return;
    if (typeof doc.removeTab !== "function") return;

    let tabs = [];
    try { tabs = doc.getTabs() || []; } catch (e) { tabs = []; }
    if (!tabs || tabs.length <= 1) return;

    const normalizedExpected = normalizeTextSafe(expectedDocName);
    let removed = 0;

    for (let i = 0; i < tabs.length; i++) {
      if ((tabs.length - removed) <= 1) break;
      const tab = tabs[i];
      if (!tab) continue;

      const tabName = normalizeTextSafe(
        (typeof tab.getTitle === "function" ? tab.getTitle() : "") ||
        (typeof tab.getName === "function" ? tab.getName() : "")
      );

      let tabBody = null;
      try {
        if (typeof tab.asDocumentTab === "function") {
          const dt = tab.asDocumentTab();
          if (dt && typeof dt.getBody === "function") tabBody = dt.getBody();
        }
      } catch (e) {}

      const bodyText = tabBody ? getBodyTextSafe(tabBody) : "";
      const bodyChildren = tabBody ? tabBody.getNumChildren() : 0;
      const isEmpty = !bodyText || bodyChildren <= 1;
      const isArtifactName =
        /^pestana\s*\d+$/i.test(tabName) ||
        /^tab\s*\d+$/i.test(tabName) ||
        /^pestana\s*sin\s*titulo$/i.test(tabName) ||
        /^untitled\s*tab$/i.test(tabName) ||
        tabName === normalizedExpected;

      if (isEmpty || isArtifactName) {
        try {
          if (typeof tab.getId === "function") {
            doc.removeTab(tab.getId());
            removed++;
          }
        } catch (e) {}
      }
    }
  }

  function getBestTemplateBody(doc) {
    let fallbackBody = doc.getBody();
    if (typeof doc.getTabs !== "function") return fallbackBody;

    let tabs = [];
    try { tabs = doc.getTabs() || []; } catch (e) { tabs = []; }
    if (!tabs || tabs.length === 0) return fallbackBody;

    let best = null;
    let bestScore = -1;

    tabs.forEach(tab => {
      try {
        if (!tab || typeof tab.asDocumentTab !== "function") return;
        const dt = tab.asDocumentTab();
        if (!dt || typeof dt.getBody !== "function") return;
        const b = dt.getBody();
        if (!b) return;

        const txt = getBodyTextSafe(b);
        const children = b.getNumChildren();
        let score = (txt ? txt.length : 0) + (children * 50);

        if (txt.includes("{{paciente}}") || txt.includes("{{fecha}}") || txt.includes("{{cedula}}")) {
          score += 5000;
        }

        if (score > bestScore) {
          bestScore = score;
          best = { tab: tab, body: b };
        }
      } catch (e) {}
    });

    if (!best) return fallbackBody;

    // Eliminamos todas las pestañas sobrantes para que no se exporten hojas extras.
    try {
      if (typeof doc.removeTab === "function" && typeof best.tab.getId === "function") {
        const keepId = best.tab.getId();
        tabs.forEach(t => {
          try {
            if (!t || typeof t.getId !== "function") return;
            const id = t.getId();
            if (id && id !== keepId) doc.removeTab(id);
          } catch (e) {}
        });
      }
    } catch (e) {}

    return best.body || fallbackBody;
  }

  const configMap = getServiceConfigDetail(data.tipo_examen);
  const CM_TO_PX = 37.7952755906; // Google Docs trabaja aproximadamente a 96dpi
  const sizeConfig = {
    small: { w: Math.round(4.32 * CM_TO_PX), h: Math.round(5.3 * CM_TO_PX), perRow: 3 },
    medium: { w: Math.round(6.57 * CM_TO_PX), h: Math.round(8.0 * CM_TO_PX), perRow: 2 },
    // Compactamos un poco mas el bloque "large" para que dos fotos seguidas entren mejor.
    large: { w: Math.round(8.7 * CM_TO_PX), h: Math.round(9.1 * CM_TO_PX), perRow: 1 }
  };
  
  // 1. OBTENER DATOS REALES (CÉDULA Y EDAD)
  let cedulaReal = data.id_paciente; 
  let edadReal = "NN";
  const pacienteInfo = findRow("pacientes", "id_paciente", data.id_paciente);
  if (pacienteInfo) {
      cedulaReal = pacienteInfo.cedula || pacienteInfo.id_paciente; 
      if (pacienteInfo.fecha_nacimiento) edadReal = calculateAgeExact(pacienteInfo.fecha_nacimiento);
  }

  // 2. PREPARAR DOC
  // Volvemos a usar plantilla para conservar membrete/imagenes/firma visual.
  // Luego dejamos solo la pestaña con contenido real para evitar hoja basura.
  const reportDocName = "REPORTE_" + data.nombre_paciente;
  const docId = DriveApp.getFileById(TEMPLATE_ID).makeCopy(reportDocName).getId();
  const doc = DocumentApp.openById(docId);
  cleanupDocTabs(doc, reportDocName);
  const body = getBestTemplateBody(doc);

  // Limpieza de variables viejas
  body.replaceText("{{titulo}}", ""); body.replaceText("{{paciente}}", "");
  body.replaceText("{{fecha}}", ""); body.replaceText("{{edad}}", ""); body.replaceText("{{medicamentos}}", "");

  // 3. Limpiar saltos de página heredados de la plantilla para evitar hoja extra
  for (let i = body.getNumChildren() - 1; i >= 0; i--) {
    const child = body.getChild(i);
    if (child.getType() === DocumentApp.ElementType.PAGE_BREAK) {
      body.removeChild(child);
    }
  }

  // 3.1 Limpiar solo artefactos de "Pestaña 1 / Tab 1" o título automático inicial
  const normalizedDocName = normalizeTextSafe(reportDocName);
  cleanupTextArtifactsInSection(body, normalizedDocName);
  try { cleanupTextArtifactsInSection(doc.getHeader(), normalizedDocName); } catch (e) {}
  try { cleanupTextArtifactsInSection(doc.getFooter(), normalizedDocName); } catch (e) {}
  for (let i = body.getNumChildren() - 1; i >= 0; i--) {
    const child = body.getChild(i);
    const txt = getElementTextSafe(child);
    if (!txt) continue;
    const isTabArtifact =
      /^pestana\s*\d+$/i.test(txt) ||
      /^tab\s*\d+$/i.test(txt) ||
      /^pestana\s*sin\s*titulo$/i.test(txt) ||
      /^untitled\s*tab$/i.test(txt) ||
      /^document\s*tab\s*\d+$/i.test(txt);
    const isAutoTitle = txt === normalizedDocName;
    if (isTabArtifact || isAutoTitle) body.removeChild(child);
  }

  // 4. CABECERA DE DATOS (TABLA INVISIBLE)
  const headerTable = body.insertTable(0);
  headerTable.setBorderWidth(0); 
  
  const r1 = headerTable.appendTableRow();
  r1.appendTableCell(`PACIENTE: ${data.nombre_paciente}`).getChild(0).asParagraph().setBold(true).setFontSize(10);
  r1.appendTableCell(`EDAD: ${edadReal}`).getChild(0).asParagraph().setBold(true).setFontSize(10).setAlignment(DocumentApp.HorizontalAlignment.RIGHT);
  
  const r2 = headerTable.appendTableRow();
  r2.appendTableCell(`C.I: ${cedulaReal}`).getChild(0).asParagraph().setBold(true).setFontSize(10);
  r2.appendTableCell(`FECHA: ${new Date().toLocaleDateString()}`).getChild(0).asParagraph().setBold(true).setFontSize(10).setAlignment(DocumentApp.HorizontalAlignment.RIGHT);

  headerTable.setColumnWidth(0, 350); headerTable.setColumnWidth(1, 150);
  body.insertParagraph(1, ""); 

  // 5. TÍTULO REAL DEL REPORTE
  const tituloReal = getRealReportTitle(data.tipo_examen) || data.tipo_examen;
  const pTitulo = body.insertParagraph(2, tituloReal.toUpperCase());
  pTitulo.setHeading(DocumentApp.ParagraphHeading.HEADING1).setAlignment(DocumentApp.HorizontalAlignment.CENTER).setForegroundColor("#36235d").setBold(true);
  pTitulo.setSpacingAfter(6);

  // 6. CONTENIDO DINÁMICO
  let compactTable = null; 
  let compactRow = null;
  let colCounter = 0;
  let imagenesImpresas = false; // BANDERA DE CONTROL

  // Helper interno para imprimir bloque de fotos respetando tamaño y orden real
  const imprimirFotos = (tituloSeccion) => {
      if (data.imagenes && data.imagenes.length > 0) {
          body.appendParagraph(tituloSeccion)
            .setHeading(DocumentApp.ParagraphHeading.HEADING3)
            .setForegroundColor("#36235d")
            .setSpacingBefore(5)
            .setSpacingAfter(2);

          const orderedImgs = data.imagenes.slice(); // Respeta orden del usuario
          let i = 0;
          while (i < orderedImgs.length) {
            const current = orderedImgs[i];
            const currentSize = sizeConfig[current.size] ? current.size : "small";
            const cfg = sizeConfig[currentSize];

            const rowTable = body.appendTable();
            rowTable.setBorderWidth(0);
            const row = rowTable.appendTableRow();

            let used = 0;
            while (i < orderedImgs.length && used < cfg.perRow) {
              const imgInfo = orderedImgs[i];
              const infoSize = sizeConfig[imgInfo.size] ? imgInfo.size : "small";
              // Si cambia de tamaño, cerramos fila para mantener orden visual correcto
              if (infoSize !== currentSize) break;

              const blob = imgsMap[imgInfo.index];
              const cell = row.appendTableCell();
              cell.setPaddingTop(1).setPaddingBottom(2).setPaddingLeft(1).setPaddingRight(1);
              cell.appendParagraph(imgInfo.title || `Foto ${imgInfo.index}`)
                .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
                .setFontSize(7)
                .setBold(true)
                .setSpacingAfter(1);

              if (blob) {
                try {
                  const img = cell.appendImage(blob);
                  const w = img.getWidth();
                  const h = img.getHeight();
                  // Llenar marco sin deformar (permitimos ampliar para evitar "grande" pequeño)
                  const scale = Math.min(cfg.w / w, cfg.h / h);
                  img.setWidth(Math.round(w * scale)).setHeight(Math.round(h * scale));
                  img.getParent().setAlignment(DocumentApp.HorizontalAlignment.CENTER);
                } catch(e) {}
              }
              used++;
              i++;
            }
            body.appendParagraph("").setSpacingAfter(currentSize === "large" ? 0 : 1);
          }
          imagenesImpresas = true;
      }
  };

  // --- RECORRIDO DE CAMPOS ---
  configMap.forEach(campo => {
    const key = campo.nombre;
    const label = campo.etiqueta;
    const type = campo.tipo;
    const valor = data.datos_json[key];

    if (!valor && type !== 'titulo' && type !== 'imagenes') return;

    // Romper tabla compacta si viene bloque grande
    if (type === 'titulo' || type === 'parrafo' || type === 'imagenes') {
        compactTable = null; compactRow = null; colCounter = 0;
    }

    // A. IMÁGENES (Si están configuradas en el orden)
    if (type === 'imagenes') {
        imprimirFotos(label.toUpperCase());
    } 
    // B. TÍTULOS
    else if (type === 'titulo') {
         body.appendParagraph(label.toUpperCase()).setHeading(DocumentApp.ParagraphHeading.HEADING3).setForegroundColor("#36235d").setSpacingBefore(12);
    }
    // C. PÁRRAFOS (Corregido: NO NEGRITA EN CONTENIDO)
    else if (type === 'parrafo') {
         // Título
         body.appendParagraph(label.toUpperCase() + ":").setBold(true).setForegroundColor("#36235d").setFontSize(10).setSpacingBefore(3);
         // Contenido (FORZAMOS setBold(false))
         body.appendParagraph(valor).setBold(false).setFontSize(10).setForegroundColor("#000000").setAlignment(DocumentApp.HorizontalAlignment.JUSTIFY).setSpacingAfter(5);
    }
    // D. DATOS CORTOS (Tabla 2 columnas)
    else {
         if (!compactTable) {
             compactTable = body.appendTable();
             compactTable.setBorderWidth(0);
             compactRow = compactTable.appendTableRow();
             colCounter = 0;
         }
         if (colCounter >= 2) {
             compactRow = compactTable.appendTableRow();
             colCounter = 0;
         }

         const cell = compactRow.appendTableCell();
         // Estilo comprimido
         const p = cell.appendParagraph(label + ": " + valor);
          p.setSpacingAfter(0);
         
         const text = p.editAsText();
         // Negrita solo a la etiqueta
         text.setBold(0, label.length, true);
         text.setForegroundColor(0, label.length, "#36235d"); 
         // Normal al valor (FORZADO)
         text.setBold(label.length + 1, text.getText().length - 1, false);
         text.setForegroundColor(label.length + 1, text.getText().length - 1, "#000000"); 

         colCounter++;
    }
  });

  // 7. SALVAVIDAS: SI LAS FOTOS NO SE IMPRIMIERON ARRIBA, IMPRIMIRLAS AL FINAL
  if (!imagenesImpresas && data.imagenes && data.imagenes.length > 0) {
      body.appendParagraph(""); // Espacio
      imprimirFotos("EVIDENCIA FOTOGRÁFICA");
  }

  appendDoctorVirtualSignature_(body, data);

  doc.saveAndClose();
  const pdfBlob = DriveApp.getFileById(docId).getAs("application/pdf");
  const pdfFile = folder.createFile(pdfBlob);
  applyFileSharingPolicy_(pdfFile);
  DriveApp.getFileById(docId).setTrashed(true);
  return pdfFile.getUrl();
}

  const c2 = row1.appendTableCell(`EDAD: ${edadReal}`);

  headerTable.setColumnWidth(1, 150);

  body.insertParagraph(1, "");

  pTitulo.setBold(true);
  pTitulo.setSpacingAfter(20);

  // ==================================================
  // BLOQUE 3: TABLA DE MEDICAMENTOS
  // ==================================================
  const table = body.appendTable();
  const header = table.appendTableRow();
  
  // Encabezados con estilo

  const h2 = header.appendTableCell("CANT");

       .setPaddingTop(5)

      c.getChild(0).asParagraph().setBold(true).setAlignment(DocumentApp.HorizontalAlignment.CENTER);

  table.setColumnWidth(0, 200); // Medicamento
  table.setColumnWidth(1, 50);  // Cantidad
  table.setColumnWidth(2, 200); // Indicaciones

  data.medicamentos.forEach(m => {
      const row = table.appendTableRow();
      row.appendTableCell(m.nombre || "").setPaddingTop(5);
      
      const cellQty = row.appendTableCell(m.cantidad.toString());
      cellQty.getChild(0).asParagraph().setAlignment(DocumentApp.HorizontalAlignment.CENTER);
      cellQty.setVerticalAlignment(DocumentApp.VerticalAlignment.MIDDLE);
      
      row.appendTableCell(m.frecuencia || "").setPaddingTop(5);
  });

  // ==================================================
  // BLOQUE 4: OBSERVACIONES
  // ==================================================
  if (data.observaciones_receta) {
      const pObs = body.appendParagraph("\nObservaciones: " + data.observaciones_receta);
      pObs.setItalic(true).setForegroundColor("#555555");
  }

  appendDoctorVirtualSignature_(body, data);

  // Guardar y Finalizar
  doc.saveAndClose();
  const pdfBlob = DriveApp.getFileById(docId).getAs("application/pdf");
  const pdfFile = folder.createFile(pdfBlob);
  applyFileSharingPolicy_(pdfFile);
  
  DriveApp.getFileById(docId).setTrashed(true);
  return pdfFile.getUrl();
}

function calculateAgeForPdf(patientId) {
    try {
        const sheet = getConnection().getSheetByName("pacientes");
        const data = sheet.getDataRange().getValues();
        for(let i=1; i<data.length; i++) {
            if(String(data[i][0]) === String(patientId)) {
                const nac = new Date(data[i][3]); // Col D
                const hoy = new Date();
                let edad = hoy.getFullYear() - nac.getFullYear();
                const m = hoy.getMonth() - nac.getMonth();
                if (m < 0 || (m === 0 && hoy.getDate() < nac.getDate())) edad--;
                return edad + " años";
            }
        }
    } catch(e) { return ""; }
    return "";
}

function getReportFolder(dataObj, isUpdate, oldJsonData, idReporte) {
    try {
        const root = DriveApp.getFoldersByName("VIDAFEM_EXPEDIENTES");
        const rootFolder = root.hasNext() ? root.next() : DriveApp.createFolder("VIDAFEM_EXPEDIENTES");
        
        let pFolder;
        const searchId = String(dataObj.id_paciente);
        const subs = rootFolder.getFolders();
        while(subs.hasNext()){
            const f = subs.next();
            if(f.getName().endsWith(searchId)) { pFolder = f; break; }
        }
        if(!pFolder) pFolder = rootFolder.createFolder(`${dataObj.nombre_paciente}_${dataObj.id_paciente}`);

        if(isUpdate && oldJsonData.report_folder_id) {

        return pFolder.createFolder(`${cleanType}_${idReporte}`);
    } catch(e) {
        return DriveApp.getRootFolder();
    }
}

function defaultDiagnosisExternalPdfLabel_(value, fallbackIndex) {
    const raw = String(value || "").trim();
    if (raw) {
        return raw.replace(/\.pdf$/i, "").trim() || raw;
    }
    return "Adjunto PDF " + String(Number(fallbackIndex || 0) + 1);
}

function normalizeDiagnosisExternalPdfItems_(items) {
    const list = Array.isArray(items) ? items : [];
    const out = [];
    for (let i = 0; i < list.length; i++) {
        const item = list[i] || {};
        const url = String(item.url || item.pdf_externo_link || "").trim();
        const fileId = String(item.file_id || item.fileId || extractDiagnosisAssetFileIdFromUrl_(url) || "").trim();
        const name = String(item.name || "").trim();
        const label = defaultDiagnosisExternalPdfLabel_(item.label || item.nombre_visible || item.display_name || name, i);
        const id = String(item.id || fileId || ("external_pdf_" + (i + 1))).trim();
        if (!url && !fileId && !name) continue;
        out.push({
            id: id,
            label: label,
            url: url,
            file_id: fileId,
            name: name
        });
    }
    return out;
}

function getLegacyDiagnosisExternalPdfItems_(payload) {
    const data = payload || {};
    const legacyUrl = String(data.pdf_externo_link || "").trim();
    if (!legacyUrl) return [];
    const fileId = extractDiagnosisAssetFileIdFromUrl_(legacyUrl);
    return [{
        id: String(fileId || "external_pdf_1").trim(),
        label: defaultDiagnosisExternalPdfLabel_(data.pdf_externo_nombre || data.titulo_adjunto || "Adjunto PDF", 0),
        url: legacyUrl,
        file_id: fileId,
        name: ""
    }];
}

function getDiagnosisExternalPdfItems_(payload) {
    const data = payload || {};
    const modern = normalizeDiagnosisExternalPdfItems_(data.pdf_externos);
    return modern.length ? modern : getLegacyDiagnosisExternalPdfItems_(data);
}

function findDiagnosisExternalPdfItemMatch_(items, probe) {
    const list = Array.isArray(items) ? items : [];
    const src = probe || {};
    const targetId = String(src.id || "").trim();
    const targetFileId = String(src.file_id || src.fileId || "").trim();
    const targetUrl = String(src.url || src.pdf_externo_link || "").trim();
    for (let i = 0; i < list.length; i++) {
        const item = list[i] || {};
        if (targetId && String(item.id || "").trim() === targetId) return item;
        if (targetFileId && String(item.file_id || item.fileId || "").trim() === targetFileId) return item;
        if (targetUrl && String(item.url || "").trim() === targetUrl) return item;
    }
    return null;
}

function trashDiagnosisExternalPdfItemSafe_(item) {
    const entry = item || {};
    const fileId = String(entry.file_id || entry.fileId || extractDiagnosisAssetFileIdFromUrl_(entry.url) || "").trim();
    if (!fileId) return;
    try { DriveApp.getFileById(fileId).setTrashed(true); } catch (e) {}
}

// Helper PDF externo
function handleExternalPdf(dataObj, isUpdate, oldJsonData, folder) {
    const oldItems = getDiagnosisExternalPdfItems_(oldJsonData);
    const requestedItems = Array.isArray(dataObj.pdf_externos) ? dataObj.pdf_externos : null;

    if (dataObj.pdf_externo && dataObj.pdf_externo.delete) {
        oldItems.forEach(trashDiagnosisExternalPdfItemSafe_);
        return [];
    }

    if (requestedItems) {
        const finalItems = [];
        requestedItems.forEach(function(item, index) {
            const current = item || {};
            if (current.delete === true) return;

            const existing = findDiagnosisExternalPdfItemMatch_(oldItems, current);
            const label = defaultDiagnosisExternalPdfLabel_(
                current.label || current.nombre_visible || current.display_name || current.name || (existing && existing.label),
                index
            );

            if (current.data) {
                try {
                    const raw = String(current.data || "");
                    const marker = raw.indexOf(",");
                    const b64 = marker > -1 ? raw.substring(marker + 1) : raw;
                    const mime = String(current.mime || "application/pdf").trim() || "application/pdf";
                    const name = String(current.name || (label + ".pdf") || ("adjunto_" + (index + 1) + ".pdf")).trim();
                    const blob = Utilities.newBlob(Utilities.base64Decode(b64), mime, name);
                    const file = folder.createFile(blob);
                    applyFileSharingPolicy_(file);
                    finalItems.push({
                        id: String(current.id || Utilities.getUuid()).trim(),
                        label: label,
                        url: file.getUrl(),
                        file_id: file.getId(),
                        name: name
                    });
                } catch (e) {}
                return;
            }

            const keptUrl = String(current.url || current.pdf_externo_link || (existing && existing.url) || "").trim();
            const keptFileId = String(current.file_id || current.fileId || (existing && existing.file_id) || extractDiagnosisAssetFileIdFromUrl_(keptUrl) || "").trim();
            if (!keptUrl) return;
            finalItems.push({
                id: String(current.id || (existing && existing.id) || keptFileId || Utilities.getUuid()).trim(),
                label: label,
                url: keptUrl,
                file_id: keptFileId,
                name: String(current.name || (existing && existing.name) || "").trim()
            });
        });

        oldItems.forEach(function(oldItem) {
            if (findDiagnosisExternalPdfItemMatch_(finalItems, oldItem)) return;
            trashDiagnosisExternalPdfItemSafe_(oldItem);
        });
        return finalItems;
    }

    if (dataObj.pdf_externo && dataObj.pdf_externo.data) {
        oldItems.forEach(trashDiagnosisExternalPdfItemSafe_);
        try {
            const blob = Utilities.newBlob(
                Utilities.base64Decode(String(dataObj.pdf_externo.data || "").split(",")[1]),
                dataObj.pdf_externo.mime,
                dataObj.pdf_externo.name
            );
            const f = folder.createFile(blob);
            applyFileSharingPolicy_(f);
            return [{
                id: String(Utilities.getUuid()).trim(),
                label: defaultDiagnosisExternalPdfLabel_(dataObj.pdf_externo.label || dataObj.pdf_externo.name || "Adjunto PDF", 0),
                url: f.getUrl(),
                file_id: f.getId(),
                name: String(dataObj.pdf_externo.name || "").trim()
            }];
        } catch (e) {
            return [];
        }
    }

    if (isUpdate) return oldItems;
    return [];
}

function diagnosisHasMeaningfulReportContent_(dataObj) {
    const data = dataObj || {};
    const fixedKeys = [
        "motivo",
        "evaluacion",
        "vagina",
        "vulva",
        "ano",
        "hallazgos",
        "diagnostico",
        "biopsia",
        "recomendaciones"
    ];
    for (let i = 0; i < fixedKeys.length; i++) {
        if (String(data[fixedKeys[i]] || "").trim()) return true;
    }

    const dyn = data.datos_json;
    if (dyn && typeof dyn === "object") {
        for (const key in dyn) {
            if (!Object.prototype.hasOwnProperty.call(dyn, key)) continue;
            if (String(dyn[key] === undefined || dyn[key] === null ? "" : dyn[key]).trim()) return true;
        }
    }

    return !!(data.imagenes && Array.isArray(data.imagenes) && data.imagenes.length);
}

function mergeDiagnosisPayloadForSave_(oldJsonData, dataObj, examType) {
    const oldData = oldJsonData || {};
    const incoming = dataObj || {};
    const mode = String(examType || incoming.tipo_examen || oldData.tipo_examen || "").trim().toUpperCase();
    const preserveBundle = mode === "RECETA" || mode === "EXAMENPDF" || mode === "TODO";
    const merged = { ...oldData, ...incoming };

    if (preserveBundle) {
        if (!Object.prototype.hasOwnProperty.call(incoming, "datos_json") && oldData.datos_json) {
            merged.datos_json = oldData.datos_json;
        }
        if (!Object.prototype.hasOwnProperty.call(incoming, "imagenes") && oldData.imagenes) {
            merged.imagenes = oldData.imagenes;
        }
        const incomingMeds = incoming.medicamentos && Array.isArray(incoming.medicamentos) ? incoming.medicamentos : [];
        if (!incomingMeds.length && oldData.medicamentos && Array.isArray(oldData.medicamentos)) {
            merged.medicamentos = oldData.medicamentos;
        }
        if (!String(incoming.observaciones_receta || "").trim() && String(oldData.observaciones_receta || "").trim()) {
            merged.observaciones_receta = oldData.observaciones_receta;
        }
    }

    merged.pdf_externos = Object.prototype.hasOwnProperty.call(incoming, "pdf_externos")
        ? normalizeDiagnosisExternalPdfItems_(incoming.pdf_externos)
        : getDiagnosisExternalPdfItems_(oldData);
    merged.pdf_externo_link = merged.pdf_externos.length
        ? String(merged.pdf_externos[0].url || "").trim()
        : "";

    return merged;
}

function extractDiagnosisAssetFileIdFromUrl_(url) {
    const raw = String(url || "").trim();
    if (!raw) return "";

    let match = raw.match(/[?&]id=([a-zA-Z0-9_-]+)/);
    if (match && match[1]) return match[1];

    match = raw.match(/\/d\/([a-zA-Z0-9_-]+)/);
    if (match && match[1]) return match[1];

    match = raw.match(/[-\w]{25,}/);
    return match && match[0] ? match[0] : "";
}

function trashDiagnosisAssetFileByUrl_(url) {
    const fileId = extractDiagnosisAssetFileIdFromUrl_(url);
    if (!fileId) return { success: false, message: "No se pudo identificar el archivo." };
    try {
        DriveApp.getFileById(fileId).setTrashed(true);
        return { success: true, fileId: fileId };
    } catch (e) {
        return { success: false, fileId: fileId, message: e.toString() };
    }
}

function deleteDiagnosisAsset(idReporte, assetType, assetId) {
    const reportId = String(idReporte || "").trim();
    const asset = String(assetType || "").trim().toLowerCase();
    const targetAssetId = String(assetId || "").trim();
    if (!reportId) return { success: false, message: "Falta id_reporte." };

    const allowed = {
        report_pdf: { label: "informe", field: "pdf_url" },
        recipe_pdf: { label: "receta", field: "pdf_receta_link" },
        external_pdf: { label: "adjunto", field: "pdf_externo_link" }
    };
    const meta = allowed[asset];
    if (!meta) return { success: false, message: "Tipo de documento no valido." };

    const sheet = getSheet("diagnosticos_archivos");
    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return { success: false, message: "No hay diagnosticos registrados." };

    const headers = data[0];
    const idxId = headers.indexOf("id_reporte");
    const idxPatient = headers.indexOf("id_paciente");
    const idxType = headers.indexOf("tipo_examen");
    const idxDate = headers.indexOf("fecha");
    const idxJson = headers.indexOf("datos_json");
    const idxPdf = headers.indexOf("pdf_url");
    const idxCreatedBy = headers.indexOf("creado_por");
    if (idxId === -1) return { success: false, message: "Estructura de diagnosticos incompleta." };

    let rowIndex = -1;
    let row = null;
    for (let i = 1; i < data.length; i++) {
        if (String(data[i][idxId] || "").trim() === reportId) {
            rowIndex = i + 1;
            row = data[i];
            break;
        }
    }
    if (rowIndex === -1 || !row) return { success: false, message: "Reporte no encontrado." };

    let jsonData = {};
    try { jsonData = JSON.parse(String(idxJson > -1 ? row[idxJson] : "") || "{}"); } catch (e) { jsonData = {}; }

    let targetUrl = "";
    let updatedPdfUrl = String(idxPdf > -1 ? row[idxPdf] || "" : "").trim();
    let externalItems = getDiagnosisExternalPdfItems_(jsonData);

    if (asset === "report_pdf") {
        targetUrl = updatedPdfUrl;
        updatedPdfUrl = "";
        if (idxPdf > -1) sheet.getRange(rowIndex, idxPdf + 1).setValue("");
    } else if (asset === "recipe_pdf") {
        targetUrl = String(jsonData.pdf_receta_link || "").trim();
        delete jsonData.pdf_receta_link;
    } else if (asset === "external_pdf") {
        if (!externalItems.length) {
            targetUrl = String(jsonData.pdf_externo_link || "").trim();
            externalItems = targetUrl ? getLegacyDiagnosisExternalPdfItems_(jsonData) : [];
        }
        if (!externalItems.length) {
            return { success: false, message: "Ese documento ya no existe o no fue generado." };
        }

        let targetIndex = 0;
        if (targetAssetId) {
            for (let i = 0; i < externalItems.length; i++) {
                const current = externalItems[i] || {};
                if (
                    String(current.id || "").trim() === targetAssetId ||
                    String(current.file_id || current.fileId || "").trim() === targetAssetId ||
                    String(current.url || "").trim() === targetAssetId
                ) {
                    targetIndex = i;
                    break;
                }
            }
        }
        const removed = externalItems.splice(targetIndex, 1)[0] || {};
        targetUrl = String(removed.url || "").trim();
        jsonData.pdf_externos = externalItems;
        jsonData.pdf_externo_link = externalItems.length ? String(externalItems[0].url || "").trim() : "";
    }

    if (!targetUrl) {
        return { success: false, message: "Ese documento ya no existe o no fue generado." };
    }

    let warningMessage = "";
    const deleteRes = trashDiagnosisAssetFileByUrl_(targetUrl);
    if (!deleteRes.success) {
        warningMessage = deleteRes.message || "No se pudo enviar el archivo a la papelera de Drive.";
    }

    if (asset !== "report_pdf" && idxJson > -1) {
        sheet.getRange(rowIndex, idxJson + 1).setValue(JSON.stringify(jsonData));
    } else if (idxJson > -1) {
        sheet.getRange(rowIndex, idxJson + 1).setValue(JSON.stringify(jsonData));
    }

    const syncRow = {
        id_reporte: reportId,
        id_paciente: idxPatient > -1 ? row[idxPatient] : "",
        tipo_examen: idxType > -1 ? row[idxType] : "",
        fecha: idxDate > -1 ? row[idxDate] : "",
        datos_json: JSON.stringify(jsonData),
        pdf_url: updatedPdfUrl,
        creado_por: idxCreatedBy > -1 ? row[idxCreatedBy] : ""
    };
    const syncRes = syncDiagnosisToSupabase_(syncRow);
    if (syncRes && !syncRes.success && !syncRes.skipped) {
        warningMessage = warningMessage
            ? (warningMessage + " | " + String(syncRes.message || "No se pudo sincronizar con Supabase."))
            : String(syncRes.message || "No se pudo sincronizar con Supabase.");
    }

    return warningMessage
        ? {
            success: true,
            message: "Documento eliminado con advertencia.",
            warning: warningMessage,
            remaining_docs: {
                pdf_url: updatedPdfUrl,
                pdf_receta_link: String(jsonData.pdf_receta_link || "").trim(),
                pdf_externo_link: String(jsonData.pdf_externo_link || "").trim(),
                pdf_externos: normalizeDiagnosisExternalPdfItems_(jsonData.pdf_externos)
            }
        }
        : {
            success: true,
            message: "Se elimino el " + meta.label + " correctamente.",
            remaining_docs: {
                pdf_url: updatedPdfUrl,
                pdf_receta_link: String(jsonData.pdf_receta_link || "").trim(),
                pdf_externo_link: String(jsonData.pdf_externo_link || "").trim(),
                pdf_externos: normalizeDiagnosisExternalPdfItems_(jsonData.pdf_externos)
            }
        };
}

  styleTitulo[DocumentApp.Attribute.FONT_SIZE] = 11;

  styleTexto[DocumentApp.Attribute.BOLD] = false;
  styleTexto[DocumentApp.Attribute.FOREGROUND_COLOR] = "#000000";
  styleTexto[DocumentApp.Attribute.FONT_SIZE] = 10;

  body.appendParagraph("\n"); // Espacio

  // Título del Examen
  body.appendParagraph((data.tipo_examen || "REPORTE").toUpperCase())
      .setHeading(DocumentApp.ParagraphHeading.HEADING2)
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
      .setAttributes(styleTitulo);

  // COLPOSCOPIA
  if (data.tipo_examen === "COLPOSCOPIA") {
      writeField(body, "EVALUACIÓN GENERAL:", data.evaluacion, styleTitulo, styleTexto);

          body.appendParagraph(`Zonas: Vagina: ${data.vagina||'-'} | Vulva: ${data.vulva||'-'} | Ano: ${data.ano||'-'}`)
              .setAttributes(styleTexto);
      }
      
      writeField(body, "HALLAZGOS:", data.hallazgos, styleTitulo, styleTexto);
      writeField(body, "DIAGNÓSTICO:", data.diagnostico, styleTitulo, styleTexto);
      writeField(body, "BIOPSIA:", data.biopsia, styleTitulo, styleTexto);
      writeField(body, "RECOMENDACIONES:", data.recomendaciones, styleTitulo, styleTexto);
  } 
  // GENERAL / OTROS
  else {
      writeField(body, "MOTIVO:", data.motivo, styleTitulo, styleTexto);
      writeField(body, "EVOLUCIÓN:", data.evaluacion, styleTitulo, styleTexto);
      
      if(data.datos_json) {
         for (const [key, value] of Object.entries(data.datos_json)) {
             if(value && key !== "tipo_examen") {
                 writeField(body, key.toUpperCase() + ":", value, styleTitulo, styleTexto);
             }
         }
      }
      writeField(body, "DIAGNÓSTICO:", data.diagnostico, styleTitulo, styleTexto);
      writeField(body, "TRATAMIENTO:", data.recomendaciones, styleTitulo, styleTexto);
  }

  // 3. FOTOS
  if(data.imagenes && imgs) {
      // Filtramos solo las que tienen blob válido
      const validImgs = data.imagenes.filter(img => imgs[img.index]);
      
      if(validImgs.length > 0) {
          body.appendParagraph("\nEVIDENCIA FOTOGRÁFICA").setAttributes(styleTitulo).setAlignment(DocumentApp.HorizontalAlignment.CENTER);
          const table = body.appendTable();
          table.setBorderWidth(0);
          let row = table.appendTableRow();
          let count = 0;

          validImgs.forEach(imgInfo => {
              if(count > 0 && count % 2 === 0) row = table.appendTableRow();
              const cell = row.appendTableCell();
              
              cell.appendParagraph(imgInfo.title || "Foto")
                  .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
                  .editAsText().setFontSize(9).setBold(true);

              try {
                  const image = cell.appendImage(imgs[imgInfo.index]);
                  image.setWidth(220).setHeight(160);
                  image.getParent().setAlignment(DocumentApp.HorizontalAlignment.CENTER);
              } catch(e) { cell.appendParagraph("(Error img)"); }

          });
      }

  try {
      body.appendParagraph("\n\n");
      const firmaBlob = DriveApp.getFileById(SIGNATURE_ID).getBlob();
      const p = body.appendParagraph("");
      p.setAlignment(DocumentApp.HorizontalAlignment.CENTER);

          .setAlignment(DocumentApp.HorizontalAlignment.CENTER)
          .editAsText().setFontSize(9);
  } catch(e) {}

  doc.saveAndClose();
  
  // 5. CONVERTIR Y DAR PERMISOS PÚBLICOS
  const pdfBlob = tempDocFile.getAs(MimeType.PDF);
  const pdfFile = folder.createFile(pdfBlob);
  applyFileSharingPolicy_(pdfFile); // Politica centralizada
  
  tempDocFile.setTrashed(true);
  return pdfFile.getUrl();
}

    }
}

      if (String(data[i][0]) === String(idReporte)) { // Columna 0 es ID
        rowIndex = i + 1; // Fila en Excel (base 1)
        pdfUrl = data[i][5]; // Columna 5 es pdf_url
        
        // --- AQUÍ LEEMOS LOS DATOS GUARDADOS PARA BUSCAR LOS IDs DE LAS FOTOS ---
        try {
            jsonData = JSON.parse(data[i][4]); 
        } catch(e) {
            jsonData = {};
        }
        break;
      }
    }

    if (rowIndex === -1) {
      return { success: false, message: "Reporte no encontrado." };
    }

    // 2. Eliminar el archivo PDF de Drive (Si existe link)
    if (pdfUrl && pdfUrl.includes("id=")) {
      try {
        const fileId = pdfUrl.match(/id=([a-zA-Z0-9_-]+)/)[1];
        if(fileId) {
          DriveApp.getFileById(fileId).setTrashed(true);
        }
      } catch (e) {
        console.log("Advertencia: No se pudo borrar el archivo de Drive o ya no existe. " + e);
      }
    }

    // 3. NUEVO: Eliminar las fotos individuales usando los IDs guardados
    if (jsonData.drive_file_ids && Array.isArray(jsonData.drive_file_ids)) {
        jsonData.drive_file_ids.forEach(fileId => {
            try {
                DriveApp.getFileById(fileId).setTrashed(true);
            } catch(e) {
                console.log("No se pudo borrar foto individual: " + fileId);
            }
        });

    sheet.deleteRow(rowIndex);

  } catch (error) {
    return { success: false, message: "Error al eliminar: " + error.toString() };

   }
}

// HELPER PARA CALCULAR EDAD EXACTA
function calculateAgeExact(birthDate) {
  if (!birthDate) return "NN";
  const birth = new Date(birthDate);
  const today = new Date();
  
  let age = today.getFullYear() - birth.getFullYear();
  const m = today.getMonth() - birth.getMonth();
  
  // Si no ha cumplido años este mes, o es el mismo mes pero no ha llegado el día
  if (m < 0 || (m === 0 && today.getDate() < birth.getDate())) {
    age--;
  }
  
  return age + " años";
}

function getServiceConfiguration() {
  const sheet = getSheet("config_campos");
  const data = sheet.getDataRange().getValues();
  
  // Si solo están los encabezados (fila 1), devolvemos vacío
  if (data.length <= 1) return {};

  const config = {};

  // Empezamos en fila 1 (la 0 son los títulos: servicio, campo_nombre, campo_etiqueta, tipo)
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    
    // Obtenemos los valores de las columnas A, B, C, D
    const servicio = String(row[0]).trim();       // Ej: "Cultivo"
    const nombreCampo = String(row[1]).trim();    // Ej: "tipo_muestra"
    const etiquetaCampo = String(row[2]).trim();  // Ej: "Tipo de Muestra:"
    const tipoCampo = String(row[3]).trim().toLowerCase(); // Ej: "texto", "parrafo", "titulo"

    // Si la fila tiene servicio y nombre de campo, la procesamos
    if (servicio && nombreCampo) {
      
      // Si es la primera vez que vemos este servicio, creamos su lista vacía
      if (!config[servicio]) {
        config[servicio] = [];
      }

      // Agregamos el campo a la lista de ese servicio
      config[servicio].push({
        nombre: nombreCampo,
        etiqueta: etiquetaCampo,
        tipo: tipoCampo
      });
    }
  }

  // Devolvemos el "Menú" completo al Frontend
  // Ejemplo: { "Cultivo": [{nombre: "muestra", ...}], "VPH": [{...}] }
  return config;
}

// Agregar un nuevo servicio con sus campos
// Database.gs - FUNCIÓN CORREGIDA

function saveServiceFull(data, requester) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);

    if (!data || !data.nombre_servicio) {
      return { success: false, message: "Falta nombre del servicio" };
    }

    const req = String(requester || "").trim().toLowerCase();
    const isSuper = (typeof isSuperadminUser === "function") ? isSuperadminUser(req) : false;
    const ensured = ensureServicesSheetHeaders_();
    const sheetServ = ensured.sheet;
    const headers = ensured.headers;
    const sheetCampos = ensureServiceConfigSheetHeaders_().sheet;

    const idxId = headers.indexOf("id");
    const idxName = headers.indexOf("nombre_servicio");
    const idxRecs = headers.indexOf("recomendaciones");
    const idxTitle = headers.indexOf("titulo_reporte");
    const idxScope = headers.indexOf("scope_visibility");
    const idxOwner = headers.indexOf("owner_usuario");
    const idxDuration = headers.indexOf("duracion_minutos");

    const isCreate = !String(data.originalName || "").trim();
    const searchName = String(data.originalName || data.nombre_servicio).trim();
    const targetName = String(data.nombre_servicio || "").trim();
    const searchNameLower = searchName.toLowerCase();
    const targetNameLower = targetName.toLowerCase();

    const dataServ = sheetServ.getDataRange().getValues();
    let rowIndex = -1;
    for (let i = 1; i < dataServ.length; i++) {
      if (String(dataServ[i][idxName] || "").trim().toLowerCase() === searchNameLower) {
        rowIndex = i + 1;
        break;
      }
    }

    for (let i = 1; i < dataServ.length; i++) {
      const rowName = String(dataServ[i][idxName] || "").trim().toLowerCase();
      if (!rowName || rowName !== targetNameLower) continue;
      if (rowIndex !== -1 && (i + 1) === rowIndex) continue;
      return { success: false, message: "Ya existe un servicio con ese nombre." };
    }

    let currentScope = "ALL";
    let currentOwner = "";
    let currentDuration = 30;
    let idServicio = "SERV-" + new Date().getTime();

    if (rowIndex !== -1) {
      const row = dataServ[rowIndex - 1];
      idServicio = String(row[idxId] || idServicio);
      currentScope = String(row[idxScope] || "ALL").toUpperCase();
      if (currentScope !== "OWNER" && currentScope !== "ALL") currentScope = "ALL";
      currentOwner = String(row[idxOwner] || "").trim().toLowerCase();
      currentDuration = idxDuration > -1 ? normalizeDurationMinutes_(row[idxDuration]) : 30;

      // Seguridad: si el servicio no tiene owner, ningun medico puede apropiarselo.
      // Solo superadmin puede asignar propietario explicitamente.
      if (!currentOwner && !isSuper) {
        return { success: false, message: "Servicio sin propietario. Solicita asignacion por superadmin." };
      }
      if (!currentOwner && isSuper) {
        currentOwner = String(data.owner_usuario || req || "").trim().toLowerCase();
      }
      if (currentOwner && currentOwner !== req && !isSuper) {
        return { success: false, message: "Solo el medico creador puede modificar este servicio." };
      }
    }

    let finalScope = currentScope;
    let finalOwner = currentOwner;
    let finalDuration = currentDuration || 30;

    if (isCreate) {
      if (!req) return { success: false, message: "No se pudo identificar al usuario creador." };
      finalScope = String(data.scope_visibility || "ALL").trim().toUpperCase();
      if (finalScope !== "OWNER" && finalScope !== "ALL") finalScope = "ALL";
      finalOwner = req;
      finalDuration = normalizeDurationMinutes_(data.duracion_minutos);
    } else {
      // En edicion se permite reconfigurar alcance, pero sin cambiar propietario.
      const requestedScope = String(data.scope_visibility || "").trim().toUpperCase();
      if (requestedScope === "OWNER" || requestedScope === "ALL") {
        finalScope = requestedScope;
      }
      if (!finalOwner && req) finalOwner = req;
      if (hasOwnKey_(data, "duracion_minutos")) {
        finalDuration = normalizeDurationMinutes_(data.duracion_minutos);
      }
    }

    const rowData = new Array(headers.length).fill("");
    rowData[idxId] = idServicio;
    rowData[idxName] = targetName;
    rowData[idxRecs] = data.recomendaciones || "";
    rowData[idxTitle] = data.titulo_reporte || "";
    rowData[idxScope] = finalScope;
    rowData[idxOwner] = finalOwner;
    if (idxDuration > -1) rowData[idxDuration] = finalDuration;

    if (rowIndex !== -1) {
      sheetServ.getRange(rowIndex, 1, 1, headers.length).setValues([rowData]);
    } else {
      sheetServ.appendRow(rowData);
    }

    const dataCampos = sheetCampos.getDataRange().getValues();
    for (let i = dataCampos.length - 1; i >= 1; i--) {
      const servEnFila = String(dataCampos[i][0] || "").trim();
      if (servEnFila === searchName || servEnFila === targetName) {
        sheetCampos.deleteRow(i + 1);
      }
    }

    if (data.campos && Array.isArray(data.campos)) {
      data.campos.forEach(campo => {
        sheetCampos.appendRow([
          targetName,
          campo.nombre,
          campo.etiqueta,
          campo.tipo,
          campo.opciones || ""
        ]);
      });
    }

    const savedService = {
      id: idServicio,
      nombre_servicio: targetName,
      recomendaciones: data.recomendaciones || "",
      titulo_reporte: data.titulo_reporte || "",
      scope_visibility: finalScope,
      owner_usuario: finalOwner,
      duracion_minutos: finalDuration,
      duracion_label: durationMinutesToLabel_(finalDuration)
    };

    const syncRes = syncServiceFullToSupabase_(savedService, data.campos || [], [searchName]);
    if (syncRes && !syncRes.success && !syncRes.skipped) {
      return {
        success: true,
        message: "Servicio guardado y configurado correctamente. Advertencia: no se pudo sincronizar con Supabase.",
        warning: syncRes.message || "No se pudo sincronizar con Supabase.",
        data: savedService
      };
    }

    return {
      success: true,
      message: "Servicio guardado y configurado correctamente.",
      data: savedService
    };

  } catch (e) {
    return { success: false, message: "Error DB: " + e.toString() };
  } finally {
    lock.releaseLock();
  }
}

function normalizeBulkIds_(ids) {
  if (!ids || !Array.isArray(ids)) return [];
  const map = {};
  const result = [];
  ids.forEach(function(id) {
    const clean = String(id === undefined || id === null ? "" : id).trim();
    if (!clean || map[clean]) return;
    map[clean] = true;
    result.push(clean);
  });
  return result;
}

function deleteAppointmentsBulk(idList) {
  const ids = normalizeBulkIds_(idList);
  if (!ids.length) {
    return { success: false, deleted_count: 0, missing_ids: [], message: "No se recibieron citas para eliminar." };
  }

  const sheet = getSheet("citas");
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) {
    return { success: false, deleted_count: 0, missing_ids: ids, message: "No hay citas registradas." };
  }

  const idxId = data[0].indexOf("id_cita");
  if (idxId === -1) {
    return { success: false, deleted_count: 0, missing_ids: ids, message: "La hoja de citas no tiene columna id_cita." };
  }

  const pending = {};
  ids.forEach(function(id) { pending[id] = true; });

  let deleted = 0;
  const deletedIds = [];
  for (let i = data.length - 1; i >= 1; i--) {
    const rowId = String(data[i][idxId] || "").trim();
    if (!pending[rowId]) continue;
    sheet.deleteRow(i + 1);
    deleted++;
    deletedIds.push(rowId);
    delete pending[rowId];
  }

  const missing = Object.keys(pending);
  const response = {
    success: deleted > 0,
    deleted_count: deleted,
    missing_ids: missing,
    message: deleted > 0
      ? "Se eliminaron " + deleted + " cita(s)."
      : "No se encontraron citas para eliminar."
  };
  const warnings = [];
  deletedIds.forEach(function(id) {
    const syncRes = deleteAppointmentFromSupabase_(id);
    if (syncRes && !syncRes.success && !syncRes.skipped) {
      warnings.push(String(syncRes.message || "No se pudo sincronizar con Supabase."));
    }
  });
  if (warnings.length) {
    response.warning = warnings.join(" | ");
    response.message = response.success
      ? ("Se eliminaron " + deleted + " cita(s). Advertencia: no se pudo sincronizar con Supabase.")
      : "No se encontraron citas para eliminar. Advertencia: no se pudo sincronizar con Supabase.";
  }
  return response;
}

// Asegurarnos de que getFolder también exista si se llama por fuera
function getFolder(name, parentFolder) {
  const parent = parentFolder || DriveApp.getRootFolder();
  const folders = parent.getFoldersByName(name);

function deleteServiceFull(nombreServicio, requester) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);

    const serviceName = String(nombreServicio || "").trim();
    if (!serviceName) return { success: false, message: "Nombre de servicio invalido." };

    const ensured = ensureServicesSheetHeaders_();
    const sheetServ = ensured.sheet;

    const dataServ = sheetServ.getDataRange().getValues();

    const idxId = headers.indexOf("id");

    let scope = "ALL";
    let owner = "";
    let serviceId = "";
    for (let i = 1; i < dataServ.length; i++) {
      if (String(dataServ[i][idxName] || "").trim() === serviceName) {
        rowIndex = i + 1;
        serviceId = idxId > -1 ? String(dataServ[i][idxId] || "").trim() : "";
        scope = idxScope > -1 ? String(dataServ[i][idxScope] || "").trim().toUpperCase() : "ALL";
        if (scope !== "OWNER" && scope !== "ALL") scope = "ALL";
        owner = idxOwner > -1 ? String(dataServ[i][idxOwner] || "").trim().toLowerCase() : "";
        break;
      }
    }
    if (rowIndex === -1) return { success: false, message: "Servicio no encontrado." };

    const access = resolveServiceDoctorRequester_(requester);
    const requesterDoctor = String(access.doctor || "").trim().toLowerCase();
    if (!owner && !access.superadmin) {
      return { success: false, message: "Servicio sin propietario. Solo superadmin puede eliminarlo." };
    }
    if (owner && !access.superadmin && (!requesterDoctor || requesterDoctor !== owner)) {
      return { success: false, message: "Solo el medico creador puede eliminar este servicio." };
    }

    const sheetCampos = ensureServiceConfigSheetHeaders_().sheet;
    const dataCampos = sheetCampos.getDataRange().getValues();

function ensurePatientEvolutionSheet_() {
  const ss = getConnection();
  const sheetName = "evolucion_paciente";
  const required = [
    "id_evolucion",
    "id_paciente",
    "fecha_consulta",
    "motivo_consulta",
    "evolucion",
    "diagnostico",
    "tratamiento",
    "sugerencias",
    "creado_por",
    "fecha_actualizacion"
  ];

  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.appendRow(required);
    return { sheet: sheet, headers: required };
  }

  if (sheet.getLastRow() === 0) {
    sheet.appendRow(required);
    return { sheet: sheet, headers: required };
  }

  const lastCol = Math.max(sheet.getLastColumn(), required.length);
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  required.forEach(function(name) {
    if (headers.indexOf(name) !== -1) return;
    const col = sheet.getLastColumn() + 1;
    sheet.getRange(1, col).setValue(name);
    headers.push(name);
  });

  return { sheet: sheet, headers: headers };
}

function parsePatientEvolutionDateTime_(value) {
  if (!value) return null;
  if (Object.prototype.toString.call(value) === "[object Date]") {
    return isNaN(value.getTime()) ? null : value;
  }

  const raw = String(value || "").trim();
  if (!raw) return null;

  const match = raw.match(/^(\d{4})-(\d{2})-(\d{2})[T\s](\d{2}):(\d{2})(?::(\d{2}))?/);
  if (match) {
    const year = Number(match[1]);
    const month = Number(match[2]);
    const day = Number(match[3]);
    const hour = Number(match[4]);
    const minute = Number(match[5]);
    const second = Number(match[6] || 0);
    const parsed = new Date(year, month - 1, day, hour, minute, second, 0);
    if (!isNaN(parsed.getTime())) return parsed;
  }

  const fallback = new Date(raw);
  return isNaN(fallback.getTime()) ? null : fallback;
}

function mapPatientEvolutionEntry_(headers, row) {
  function valueOf(name) {
    const idx = headers.indexOf(name);
    return idx > -1 ? row[idx] : "";
  }

  const id = String(valueOf("id_evolucion") || "").trim();
  if (!id) return null;

  return {
    id_evolucion: id,
    id_paciente: String(valueOf("id_paciente") || "").trim(),
    fecha_consulta: valueOf("fecha_consulta") || "",
    motivo_consulta: String(valueOf("motivo_consulta") || "").trim(),
    evolucion: String(valueOf("evolucion") || "").trim(),
    diagnostico: String(valueOf("diagnostico") || "").trim(),
    tratamiento: String(valueOf("tratamiento") || "").trim(),
    sugerencias: String(valueOf("sugerencias") || "").trim(),
    creado_por: String(valueOf("creado_por") || "").trim(),
    fecha_actualizacion: valueOf("fecha_actualizacion") || ""
  };
}

function getPatientEvolutionEntries(patientId) {
  const ensured = ensurePatientEvolutionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];

  const list = [];
  for (let i = 1; i < data.length; i++) {
    const mapped = mapPatientEvolutionEntry_(headers, data[i]);
    if (!mapped) continue;
    if (String(mapped.id_paciente) !== String(patientId)) continue;
    list.push(mapped);
  }

  return list.sort(function(a, b) {
    const aTime = new Date(a.fecha_consulta || a.fecha_actualizacion || 0).getTime() || 0;
    const bTime = new Date(b.fecha_consulta || b.fecha_actualizacion || 0).getTime() || 0;
    return bTime - aTime;
  });
}

function savePatientEvolutionEntry(dataObj, requester) {
  const data = dataObj || {};
  const patientId = String(data.id_paciente || "").trim();
  const requesterDoctor = String(requester || "").trim().toLowerCase();
  if (!patientId || !requesterDoctor) {
    return { success: false, message: "Faltan datos para guardar la evolucion." };
  }

  const ensured = ensurePatientEvolutionSheet_();
  const sheet = ensured.sheet;
  const headers = ensured.headers;
  const values = sheet.getDataRange().getValues();

  const idxId = headers.indexOf("id_evolucion");
  const idxPatient = headers.indexOf("id_paciente");
  const idxDate = headers.indexOf("fecha_consulta");
  const idxReason = headers.indexOf("motivo_consulta");
  const idxProgress = headers.indexOf("evolucion");
  const idxDiagnosis = headers.indexOf("diagnostico");
  const idxTreatment = headers.indexOf("tratamiento");
  const idxSuggestions = headers.indexOf("sugerencias");
  const idxOwner = headers.indexOf("creado_por");
  const idxUpdated = headers.indexOf("fecha_actualizacion");

  const targetId = String(data.id_evolucion || "").trim();
  let rowIndex = -1;
  let createdAt = parsePatientEvolutionDateTime_(data.fecha_consulta) || new Date();
  let createdBy = requesterDoctor;
  let storedPatientId = patientId;

  if (targetId) {
    for (let i = 1; i < values.length; i++) {
      if (String(values[i][idxId] || "").trim() === targetId) {
        rowIndex = i + 1;
        storedPatientId = String(values[i][idxPatient] || patientId).trim() || patientId;
        createdAt = parsePatientEvolutionDateTime_(data.fecha_consulta) || values[i][idxDate] || new Date();
        createdBy = String(values[i][idxOwner] || requesterDoctor).trim().toLowerCase();
        break;
      }
    }
    if (rowIndex === -1) {
      return { success: false, message: "La evolucion que intentas editar no existe." };
    }
  }

  const idEvolution = targetId || ("EVOL-" + new Date().getTime());
  const row = new Array(headers.length).fill("");
  row[idxId] = idEvolution;
  row[idxPatient] = storedPatientId;
  row[idxDate] = createdAt;
  row[idxReason] = data.motivo_consulta || "";
  row[idxProgress] = data.evolucion || "";
  row[idxDiagnosis] = data.diagnostico || "";
  row[idxTreatment] = data.tratamiento || "";
  row[idxSuggestions] = data.sugerencias || "";
  row[idxOwner] = createdBy || requesterDoctor;
  row[idxUpdated] = new Date();

  if (rowIndex > 0) {
    sheet.getRange(rowIndex, 1, 1, headers.length).setValues([row]);
  } else {
    sheet.appendRow(row);
  }

  const response = {
    success: true,
    message: "Evolucion guardada correctamente.",
    data: mapPatientEvolutionEntry_(headers, row)
  };
  const syncRes = syncPatientEvolutionToSupabase_(response.data || {});
  if (syncRes && !syncRes.success && !syncRes.skipped) {
    response.warning = syncRes.message || "No se pudo sincronizar con Supabase.";
    response.message = "Evolucion guardada correctamente. Advertencia: no se pudo sincronizar con Supabase.";
  }
  return response;
}

function getInactivePatients(requester) {
  const req = String(requester || "").trim().toLowerCase();
  const isSuper = (typeof isSuperadminUser === "function" && isSuperadminUser(req));
  const pSheet = getSheet("pacientes");
  const pData = pSheet.getDataRange().getValues();
  if (pData.length <= 1) return { success: true, data: [] };
  
  const pH = pData[0];
  const pIdIdx = pH.indexOf("id_paciente");
  const pNameIdx = pH.indexOf("nombre_completo");
  const pCedIdx = pH.indexOf("cedula");
  const pRegIdx = pH.indexOf("fecha_registro");
  const pCreatorIdx = pH.indexOf("creado_por");
  
  const patients = [];
  const activityMap = {};
  
  for (let i = 1; i < pData.length; i++) {
      const row = pData[i];
      const creator = String(row[pCreatorIdx] || "").trim().toLowerCase();
      if (!isSuper && creator !== req) continue;
      const pid = String(row[pIdIdx] || "").trim();
      if (!pid) continue;
      const regDate = new Date(toIsoDate(row[pRegIdx])).getTime();
      activityMap[pid] = isNaN(regDate) ? 0 : regDate;
      patients.push({ id_paciente: pid, cedula: String(row[pCedIdx] || "").trim(), nombre_completo: String(row[pNameIdx] || "").trim(), fecha_registro: toIsoDate(row[pRegIdx]), creado_por: creator });
  }
  if (!patients.length) return { success: true, data: [] };
  const processTable = function(sheetName, dateCols, idCol) {
      const sheet = getConnection().getSheetByName(sheetName);
      if (!sheet) return;
      const data = sheet.getDataRange().getValues();
      if (data.length <= 1) return;
      const h = data[0];
      const idIdx = h.indexOf(idCol || "id_paciente");
      const dateIdxs = dateCols.map(function(c) { return h.indexOf(c); }).filter(function(i) { return i !== -1; });
      if (idIdx === -1 || !dateIdxs.length) return;
      for (let i = 1; i < data.length; i++) {
          const row = data[i];
          const pid = String(row[idIdx] || "").trim();
          if (!pid || activityMap[pid] === undefined) continue;
          dateIdxs.forEach(function(idx) { const d = new Date(toIsoDate(row[idx])).getTime(); if (!isNaN(d) && d > activityMap[pid]) { activityMap[pid] = d; } });
      }
  };
  processTable("citas", ["fecha"]);
  processTable("diagnosticos_archivos", ["fecha"]);
  processTable("evolucion_paciente", ["fecha_consulta", "fecha_actualizacion"]);
  const threshold = Date.now() - (90 * 24 * 60 * 60 * 1000);
  const inactive = [];
  patients.forEach(function(p) { const lastAct = activityMap[p.id_paciente]; if (lastAct < threshold) { p.last_activity = lastAct > 0 ? new Date(lastAct).toISOString().split("T")[0] : "Sin fecha"; inactive.push(p); } });
  inactive.sort(function(a, b) { return String(a.last_activity).localeCompare(String(b.last_activity)); });
  return { success: true, data: inactive };
}
